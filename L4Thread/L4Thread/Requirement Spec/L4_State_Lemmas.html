section‹SetError›
subsection‹memory›
lemma SetError_direct_eq:"(SetError s gno error)⊢x↝⇧1y = s⊢x↝⇧1y"
  by (auto simp add:direct_path_def SetError_def SetThreadError_def)

lemma SetError_tran1:"s⊢x↝⇧+y ⟹ (SetError s gno error)⊢x↝⇧+y"
proof(induct rule:tran_path.induct)
  case (one_path x y)
  thm one_path.hyps
  then have "(SetError s gno error)⊢x↝⇧1y"
    using SetError_direct_eq by simp
  then show ?case
    by(rule tran_path.intros)
next
  case (tran_path x y z)
  then have h2:"s⊢x↝⇧1y ∧ s⊢y↝⇧+z" by simp
    have h3: "(SetError s gno error)⊢y↝⇧+z"
    by (simp add: tran_path.hyps(3))
  then have h21:"(SetError s gno error)⊢x↝⇧1y"
    using SetError_direct_eq h2 by simp
  then show ?case
    using h3 tran_path.intros(2) by blast
qed

lemma SetError_tran2:"(SetError s gno error)⊢x↝⇧+y ⟹ s⊢x↝⇧+y"
proof(induct rule:tran_path.induct)
  case (one_path x y)
  then show ?case
    using SetError_direct_eq tran_path.intros by auto
next
  case (tran_path x y z)
  then show ?case using SetError_direct_eq tran_path.intros by auto
qed

lemma SetError_tran:"(SetError s gno error)⊢x↝⇧+y = s⊢x↝⇧+y"
  using SetError_tran1 SetError_tran2
  by blast

lemma SetError_rtran:"(SetError s gno error)⊢x↝⇧*y = s⊢x↝⇧*y"
  using SetError_tran rtran_tran
  by auto

lemma SetError_NC_User:
"gno ∉ active_threads s ⟶ 
  thread_error (SetError s gno error) = thread_error s ∧
  thread_pager (SetError s gno error) = thread_pager s ∧
  thread_handler (SetError s gno error) = thread_handler s ∧
  thread_message (SetError s gno error) = thread_message s ∧
  thread_rcvWindow (SetError s gno error) = thread_rcvWindow s"
  by(auto simp add: SetError_def SetThreadError_def)

lemma SetError_NC_User':
" thread_pager (SetError s gno error) = thread_pager s ∧
  thread_handler (SetError s gno error) = thread_handler s ∧
  thread_message (SetError s gno error) = thread_message s ∧
  thread_rcvWindow (SetError s gno error) = thread_rcvWindow s"
  by(auto simp add: SetError_def SetThreadError_def)

lemma SetError_NC_User_Other:
"(gno ≠ g ⟶
  thread_pager (SetError s gno error) = thread_pager s ∧
  thread_handler (SetError s gno error) = thread_handler s ∧
  thread_message (SetError s gno error) = thread_message s ∧
  thread_rcvWindow (SetError s gno error) = thread_rcvWindow s ∧
  thread_error (SetError s gno error) g = thread_error s g)"
  by(auto simp add: SetError_def SetThreadError_def)

lemma SetError_C_User:
"(gno ∈ active_threads s ∧ gno ∉ kIntThreads ⟶
  thread_pager (SetError s gno error) gno = thread_pager s gno ∧
  thread_handler (SetError s gno error) gno = thread_handler s gno ∧
  thread_message (SetError s gno error) gno = thread_message s gno ∧
  thread_rcvWindow (SetError s gno error) gno = thread_rcvWindow s gno ∧
  thread_error (SetError s gno error) gno = Some error)"
  by(auto simp add: SetError_def SetThreadError_def)
