theory L4_Machine
  imports L4_Init
begin

section‹ Instantiation and Its Proofs of Security Model  ›

consts SysConf :: "Sys_Config" 

definition sys_config_witness :: Sys_Config 
where 
"sys_config_witness ≡ ⦇ MaxThreadsPerSpace = 6, 
                        Threads_Gno = {kSigma0, kRootServer} ∪ kIntThreads,
                        Address_Space = {Sigma0Space,RootServerSpace,KernelSpace} ⦈" 
lemma spaceconflemma:"Sigma0Space ∈ Address_Space sys_config_witness ∧
 KernelSpace ∈ Address_Space sys_config_witness ∧
RootServerSpace ∈ Address_Space sys_config_witness ∧ Sigma0Space ≠ KernelSpace ∧
              Sigma0Space ≠ RootServerSpace ∧ KernelSpace ≠ RootServerSpace ∧ MaxAddressSpaces sys_config_witness ≥3"
  apply (simp add: sys_config_witness_def MaxAddressSpaces_def)
  using SpaceDiff by auto

lemma threadconflemma:"idle ∉ Threads_Gno sys_config_witness ∧ kSigma0 ∈ Threads_Gno sys_config_witness ∧ kRootServer ∈ Threads_Gno sys_config_witness ∧
kIntThreads ≠ {} ∧ kSigma0 ≠ kRootServer ∧ kSigma0 ∉ kIntThreads ∧ kRootServer ∉ kIntThreads ∧
 card kIntThreads ≤ MaxThreadsPerSpace sys_config_witness ∧ MaxThreads sys_config_witness ≥ 2 "
  apply (simp add:IntThreadsNotNull sys_config_witness_def MaxThreads_def ThreadIDDiff IdleDiff)
  using IntThreadsNotNull card.infinite card_mono finite_insert le_0_eq subset_insertI subset_insertI2
  by (smt dual_order.trans numeral_le_iff semiring_norm(68) semiring_norm(72))

lemma  intthreadconflemma:" kIntThreads ⊂ Threads_Gno sys_config_witness "
  using sys_config_witness_def ThreadIDDiff
  by (metis (no_types, lifting) Sys_Config.select_convs(2) Un_insert_left insert_iff sup.right_idem sup.strict_order_iff)

specification (SysConf)
  PriSpacesConfigLemma: "RootServerSpace ∈ Address_Space SysConf ∧
  Sigma0Space ∈ Address_Space SysConf ∧ KernelSpace ∈ Address_Space SysConf ∧
  Sigma0Space ≠ KernelSpace ∧Sigma0Space ≠ RootServerSpace ∧ KernelSpace ≠ RootServerSpace"
  PriThreadsConfigLemma: "idle ∉ Threads_Gno SysConf ∧ kSigma0 ∈ Threads_Gno SysConf ∧ kRootServer ∈ Threads_Gno SysConf ∧
  kIntThreads ⊂ Threads_Gno SysConf ∧ kIntThreads ≠ {} ∧ card kIntThreads ≤ MaxThreadsPerSpace SysConf ∧ 
  kSigma0 ≠ kRootServer ∧ kSigma0 ∉ kIntThreads ∧ kRootServer ∉ kIntThreads"
  KIPConfigLemma: "MaxThreads SysConf ≥ 2 ∧ MaxAddressSpaces SysConf ≥ 3"
  using sys_config_witness_def spaceconflemma threadconflemma intthreadconflemma
  by auto

consts s0_req :: State
definition s0_req_witness :: State
  where "s0_req_witness ≡ System_Init SysConf"

specification (s0_req) 
  S0ReqInit: "s0_req = System_Init SysConf"
  by simp

definition step::"State ⇒ Event ⇒ State" where
"step S e ≡
(case e of
            (eThreadControl destNo spaceSpec schedNo pagerNo)
                  ⇒ ThreadControl SysConf S destNo spaceSpec schedNo pagerNo)"

primrec run::"State ⇒ Event list ⇒ State" where
"run s [] = s" |
"run s (e#es) = run (step s e) es"

definition "Reachable s ≡ ∃es. run s0_req es = s"

lemma run_step:"∀s. run s (es @ [e]) = step (run s es) e"
  apply(induction es)
  by auto

lemma Reach_step:"Reachable s ⟹ Reachable (step s e)"
  using run_step unfolding Reachable_def
  by auto

end
