theory L4_WeakSyscall
  imports L4_Thread
begin

definition WeakCreateThread_Cond::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ bool" where
"WeakCreateThread_Cond SysConf s destNo spaceSpecifier schedNo pagerNo ≡
    (current_thread s ∈ active_threads s) ∧ (dIsPrivilegedSpace (GetCurrentSpace s)) ∧ 
    (destNo ∈ GLOBAL_GNO SysConf) ∧ (destNo ∉ GetThreadsTids s) ∧ (spaceSpecifier ∈ GLOBAL_GNO SysConf) ∧ 
    (schedNo ∈ GetThreadsTids s) ∧ (pagerNo ∈ (GetThreadsTids s ∪ {NILTHREAD})) ∧ (threads s ≠ Threads_Gno SysConf) ∧
    (spaceSpecifier = destNo ⟶ spaces s ≠ Address_Space SysConf) ∧
    (spaceSpecifier ≠ destNo ⟶ (spaceSpecifier ∈ GetThreadsTids s) ∧ GetThreadSpace s (TidToGno spaceSpecifier) ∈ spaces s) ∧
    ((pagerNo ∈ GetThreadsTids s) ⟶ (spaceSpecifier ∈ GetThreadsTids s) ∧ (GetThreadSpace s (TidToGno spaceSpecifier) ∈ initialised_spaces s) ∧ 
                                        (TidToGno schedNo ∈ active_threads s)) ∧
    ((spaceSpecifier ∈ GetThreadsTids s) ⟶ 
                (GetSpaceThreadsNums s (GetThreadSpace s (TidToGno spaceSpecifier)) < MaxThreadsPerSpace SysConf))"

definition WeakCreateThread::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t  ⇒ threadid_t ⇒ threadid_t ⇒ State" where
"WeakCreateThread SysConf s destNo spaceSpecifier schedNo pagerNo = 
(if WeakCreateThread_Cond SysConf s destNo spaceSpecifier schedNo pagerNo
 then (let gno = TidToGno destNo
        in (if (pagerNo = NILTHREAD) 
            then (if (spaceSpecifier = destNo)
                  then (let space = (SOME sp. sp ∈ (Address_Space SysConf - spaces s))
                        in CreateThread SysConf s gno space (TidToGno schedNo))
                  else  CreateThread SysConf s gno (GetThreadSpace s (TidToGno spaceSpecifier)) (TidToGno schedNo)) 
            else (if(spaceSpecifier = destNo)
                  then s                                                                             ― ‹It is impossible›
                  else CreateActiveThread SysConf s gno (GetThreadSpace s (TidToGno spaceSpecifier)) (TidToGno schedNo) (TidToGno pagerNo))))
 else s)"

definition WeakModifyThread_Cond::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ bool" where
"WeakModifyThread_Cond SysConf s destNo spaceSpecifier schedNo pagerNo ≡ 
    (current_thread s ∈ active_threads s) ∧ (dIsPrivilegedSpace (GetCurrentSpace s)) ∧ 
    (destNo ∈ GetThreadsTids s) ∧ (spaceSpecifier ∈ GetThreadsTids s) ∧
    (schedNo ∈ GetThreadsTids s ∪ {NILTHREAD}) ∧ (pagerNo ∈ GetThreadsTids s ∪ {NILTHREAD}) ∧
    ((pagerNo ≠ NILTHREAD) ⟶ GetThreadSpace s (TidToGno spaceSpecifier) ∈ initialised_spaces s) ∧
    ((pagerNo ≠ NILTHREAD) ∧ (destNo ∉ GetActiveThreadsTids s) ⟶ 
            ((schedNo ≠ NILTHREAD) ⟶ (TidToGno schedNo ∈ active_threads s)) ∧ 
            ((schedNo = NILTHREAD) ⟶ (GetThreadScheduler s (TidToGno destNo) ∈ active_threads s))) ∧
    (GetThreadSpace s (TidToGno spaceSpecifier) ≠ GetThreadSpace s (TidToGno destNo) ⟶
            GetSpaceThreadsNums s (GetThreadSpace s (TidToGno spaceSpecifier)) < MaxThreadsPerSpace SysConf)"

definition WeakModifyThread::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ State" where
"WeakModifyThread SysConf s destNo spaceSpecifier schedNo pagerNo = 
(if WeakModifyThread_Cond SysConf s destNo spaceSpecifier schedNo pagerNo
 then let tmp_dest = TidToGno destNo;
          tmp_spec = TidToGno spaceSpecifier;
          tmp_specSpace = GetThreadSpace s tmp_spec;
          tmp_destSpace = GetThreadSpace s tmp_dest in
      (if tmp_specSpace = tmp_destSpace ∧ pagerNo = NILTHREAD ∧ schedNo = NILTHREAD 
         then s                   ― ‹skip›
         else (let S1 = (if tmp_specSpace ≠ tmp_destSpace
                         then Migrate SysConf s tmp_dest tmp_specSpace
                         else s);
                   S2 = (if pagerNo ≠ NILTHREAD
                         then (if tmp_dest ∉ active_threads S1
                               then ActivateThread SysConf S1 tmp_dest (TidToGno pagerNo)
                               else SetPager S1 tmp_dest (TidToGno pagerNo))
                         else S1);
                   S3 = (if schedNo ≠ NILTHREAD
                         then SetScheduler S2 tmp_dest (TidToGno schedNo)
                         else S2)
               in S3))
 else s)"

definition WeakDeleteThread_Cond::"Sys_Config ⇒ State ⇒ threadid_t ⇒ bool" where
"WeakDeleteThread_Cond SysConf s dest ≡
    (current_thread s ∈ active_threads s) ∧ (dIsPrivilegedSpace (GetCurrentSpace s)) ∧ 
    (dest ∈ GetThreadsTids s) ∧ (thread_space s (TidToGno dest) ≠ Some Sigma0Space) ∧
    (thread_space s (TidToGno dest) ≠ Some RootServerSpace) ∧
    (thread_space s (TidToGno dest) ≠ Some KernelSpace)"

definition WeakDeleteThread::"Sys_Config ⇒ State ⇒ threadid_t ⇒ State" where
"WeakDeleteThread SysConf s dest =
(if WeakDeleteThread_Cond SysConf s dest
 then DeleteThread s (TidToGno dest)
 else s)"

definition IntThreadControl_Cond::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ bool" where
"IntThreadControl_Cond SysConf s destNo handlerNo ≡ 
    (current_thread s ∈ active_threads s) ∧ (dIsPrivilegedSpace (GetThreadSpace s (current_thread s))) ∧ 
    (destNo ∈ GetThreadsTids s) ∧ (handlerNo ∈ GetThreadsTids s) ∧ (TidToGno destNo ∈ kIntThreads)"

definition IntThreadControl::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ State" where
"IntThreadControl SysConf s destNo handlerNo = 
(if IntThreadControl_Cond SysConf s destNo handlerNo
 then (if destNo = handlerNo 
            then DeactivateInterrupt s (TidToGno destNo)
            else ActivateInterrupt SysConf s (TidToGno destNo) (TidToGno handlerNo))
 else s)"

definition Weak_Exregs::"Sys_Config ⇒ State ⇒ globalid_t ⇒ control_t set ⇒ globalid_t ⇒ State" where
"Weak_Exregs SysConf S gno control pager =
(if (gno ∈ active_threads S) ∧ (pager ∈ Threads_Gno SysConf) ∧ (gno ∉ kIntThreads)
 then (let S1 = (if ex_p ∈ control
                 then SetThreadPager S gno (GnoToTid pager)
                 else S);
           S2 = (if ex_S ∈ control ∧ isSending (GetThreadState S1 gno) ∨
                    ex_R ∈ control ∧ isReceiving (GetThreadState S1 gno)
                   then (let S11 = Unwind S1 gno;
                             S12 = SetThreadState S11 gno tsRunning;
                             S13 = enqueue_ready S12 gno
                          in S13)
                   else S1);
           S3 = (if ex_h ∈ control
                 then (if ex_H ∈ control
                       then (SetThreadState S2 gno tsAborted) 
                       else (SetThreadState S2 gno tsHalted))
                 else S2)
        in S3)
 else S)"

definition GetLowWord::"int ⇒ int" where
"GetLowWord w = w mod 2^16"

definition GetHighWord::"int ⇒ int" where
"GetHighWord w = w div 2^16"

definition do_schedule::"State ⇒ globalid_t ⇒ int ⇒ nat ⇒ State" where
"do_schedule s destNo timeControl prio =
(let old_prio = GetThreadPriority s destNo;
     S1 = (if prio < max_nat ∧ prio ≠ GetThreadPriority s destNo
           then let s1 = (dequeue_ready s destNo);
                 s2 = SetThreadPriority s1 destNo (prio mod (MAX_PRIO+1));
                 s3 = enqueue_ready s2 destNo in s3
           else s);            
     S2 = (if timeControl < max_int
           then let s1 = SetThreadQuantum S1 destNo (GetLowWord timeControl);
                    s2 = SetThreadTimeslice s1 destNo (GetHighWord timeControl)
                in s2
           else S1)
  in (S2))"

end
