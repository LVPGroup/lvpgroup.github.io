lemma in_threads_if_rule:"⋀s s1 s2 a g. g ∈ threads s1 ⟹  g ∈ threads s2 ⟹ g ∈ threads(if a then s1 else s2)"
  by presburger
lemma threads_if_rule:"⋀s s1 s2 a. threads s1 ≡ threads s ⟹  threads s2 ≡ threads s ⟹ threads(if a then s1 else s2) ≡ threads s "
  by presburger
lemma active_threads_if_rule:"⋀s s1 s2 a. active_threads s1 ≡ active_threads s ⟹  active_threads s2 ≡ active_threads s ⟹ active_threads(if a then s1 else s2) ≡ active_threads s "
  by presburger
lemma Invariants_if_rule:"⋀s s1 s2 a. Invariants s1 ⟹  Invariants s2 ⟹ Invariants(if a then s1 else s2)"
  by presburger
