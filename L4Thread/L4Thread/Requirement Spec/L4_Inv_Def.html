section‹Current›
definition Inv_Current_Thread_In_Active::"State ⇒ bool" where
"Inv_Current_Thread_In_Active s ≡
    (current_thread s = idle ∨ current_thread s ∈ active_threads s)"

definition Inv_Current_Space_IsNot_None::"State ⇒ bool" where
"Inv_Current_Space_IsNot_None s ≡ 
  (thread_space s (current_thread s) ≠ None)"

definition Inv_Current::"State ⇒ bool" where
"Inv_Current s ≡ (Inv_Current_Thread_In_Active s ∧ Inv_Current_Space_IsNot_None s)"

section‹Space›
definition Inv_Space_HasNot_Loop :: "State ⇒ bool" where
"Inv_Space_HasNot_Loop s ≡ (∀sp v1. (∄v2. s ⊢ (Virtual sp v1) ↝⇧+ (Virtual sp v2)))"

definition Inv_Space_Valid_Has_Real:: "State ⇒ bool" where
"Inv_Space_Valid_Has_Real s ≡ (∀x. (s ⊢ x ⟷ (∃r. s ⊢ x ↝⇧* (Real r))))"

definition Inv_Space_Perms_Subset::"State ⇒ bool" where
"Inv_Space_Perms_Subset s ≡
  (∀sp1 sp2 v1 v2. s ⊢ (Virtual sp1 v1) ↝⇧1 (Virtual sp2 v2) ⟶ 
        get_perms s sp1 v1 ⊆ get_perms s sp2 v2)"

definition Inv_Space_Valid_In_Spaces::"State ⇒ bool" where
"Inv_Space_Valid_In_Spaces s ≡ (∀sp v. s ⊢ (Virtual sp v) ⟶ sp ∈ spaces s)"

lemma Inv_Space_Valid_In_Spaces_lemma:
  assumes "s ⊢ (Virtual sp v)"
  shows "sp ∈ spaces s"
proof-
  have h1:"∃m. space_mapping s sp = Some m" 
    using assms valid_page_def direct_path_def by auto
  then show ?thesis unfolding spaces_def by auto
qed

definition Inv_Space_Father_Is_Unique::"State ⇒ bool" where
"Inv_Space_Father_Is_Unique s ≡ (∀x y1 y2. s ⊢ x ↝⇧1 y1 ∧ s ⊢ x ↝⇧1 y2 ⟶ y1 = y2)"

definition Inv_Space_Path_Is_Unique::"State ⇒ bool" where
"Inv_Space_Path_Is_Unique s ≡ 
 (∀sp v_page r1 r2. (s ⊢ (Virtual sp v_page) ↝⇧+ (Real r1) ∧
                    (s ⊢ (Virtual sp v_page) ↝⇧+ (Real r2)) ⟶ r1 = r2))"

lemma Space_Father_Is_Unique:"s ⊢ x ↝⇧1 y1 ∧ s ⊢ x ↝⇧1 y2 ⟶ y1 = y2"
  using direct_path_def
  by auto

lemma YIsValid:"Inv_Space_Valid_Has_Real s ⟹ s ⊢ x ↝⇧1 y ⟶ s ⊢ y"
  unfolding Inv_Space_Valid_Has_Real_def
proof-
  assume a1:"∀x. s⊢x = (∃r. s⊢x↝⇧*(Real r))"
  from this have h1:"∀x y. s⊢x↝⇧1y ⟶ s⊢y"
  proof-
    show "∀x y. s⊢x↝⇧1y ⟶ s⊢y"
      apply rule+
      apply(case_tac "∃r. y = Real r")
       apply(auto simp add:valid_page_def)
    proof-
      fix x y
      assume a11:"s⊢x↝⇧1y" and a12:"∀r. y ≠ Real r"
      from a11 have h11:"∃sp vpage. x = Virtual sp vpage" using direct_path_def by auto
      have h12:"s⊢x" using a11 FatherEqValid by auto
      have h13:"∃r. s⊢x↝⇧+(Real r)" using h11 h12 a1 rtran_tran by auto
      then obtain r where o1:"s⊢x↝⇧+(Real r)" by auto
      from this have h14:"∃z. s⊢x↝⇧1z ∧ s⊢z ↝⇧+(Real r)" 
        using a11 a12 Space_Father_Is_Unique by (meson tran_path.cases)
      then obtain z where o2:"s⊢x↝⇧1z ∧ s⊢z ↝⇧+(Real r)" by auto
      then have "s⊢y ↝⇧+(Real r)" using a11 Space_Father_Is_Unique by auto
      then have "∃z. s⊢y ↝⇧1 z" using tran_path.cases by blast
      then show "case y of Virtual sp v_page ⇒ ∃p. s⊢y↝⇧1p | Real r ⇒ True"
        using FatherIsVirtual page_t.simps(5) by blast
    qed
  qed
  then show ?thesis by simp
qed

lemma YTranIsValid:"Inv_Space_Valid_Has_Real s ⟹
 (s ⊢ x ↝⇧+ y) ⟹ s ⊢ y"
proof-
  assume a1:"Inv_Space_Valid_Has_Real s"
  show "(s ⊢ x ↝⇧+ y) ⟹ s ⊢ y"
  proof(induct rule:tran_path.induct)
    case (one_path x y)
    then show ?case using a1 YIsValid by simp
  next
    case (tran_path x y z)
    then show ?case by simp
  qed
qed

lemma tran_one_unique:"s⊢x↝⇧+y2 ⟹ s⊢x↝⇧1y ⟹
           Inv_Space_Father_Is_Unique s ⟹
           Inv_Space_HasNot_Loop s ⟹ y = Virtual sp v_page1 ⟹ y2 = Virtual sp v_page2 ⟹ v_page1 = v_page2"
  apply(induct rule:tran_path.induct)
  subgoal
    unfolding Inv_Space_Father_Is_Unique_def
    by auto
  unfolding Inv_Space_Father_Is_Unique_def
  Inv_Space_HasNot_Loop_def
  by blast

lemma tran_more_unique:"s⊢x↝⇧+y2 ⟹s⊢x↝⇧1y ⟹
       s⊢y↝⇧+z ⟹
       (s⊢y↝⇧+y2 ⟹
        Inv_Space_Father_Is_Unique s ⟹
        Inv_Space_HasNot_Loop s ⟹ z = Virtual sp v_page1 ⟹ y2 = Virtual sp v_page2 ⟹ v_page1 = v_page2) ⟹
       Inv_Space_Father_Is_Unique s ⟹
       Inv_Space_HasNot_Loop s ⟹ z = Virtual sp v_page1 ⟹ y2 = Virtual sp v_page2 ⟹ v_page1 = v_page2"
  apply(induct rule:tran_path.induct)
    unfolding Inv_Space_Father_Is_Unique_def
    Inv_Space_HasNot_Loop_def apply blast
  by blast

lemma YYIsUnique_Virtual:"(s ⊢ x ↝⇧+ y1) ⟹ (s ⊢ x ↝⇧+ y2) ⟹
Inv_Space_Father_Is_Unique s ⟹ Inv_Space_HasNot_Loop s ⟹ y1 = Virtual sp v_page1 ⟹ y2 = Virtual sp v_page2 ⟹
 v_page1 = v_page2"
  apply(induct rule:tran_path.induct)
  using tran_one_unique apply simp
  using tran_more_unique by blast


lemma tran_one_unique':"
           s⊢x↝⇧+y2 ⟹s⊢x↝⇧1y ⟹
           Inv_Space_Father_Is_Unique s ⟹ Inv_Space_HasNot_Loop s ⟹ y = Real r1 ⟹ y2 = Real r2 ⟹ r1 = r2"
  apply(induct rule:tran_path.induct)
  subgoal
    unfolding Inv_Space_Father_Is_Unique_def
    by auto
  unfolding Inv_Space_Father_Is_Unique_def
  by (metis SonIsNoReal tran_path.cases)

lemma tran_more_unique':"
s⊢x↝⇧+y2 ⟹s⊢x↝⇧1y ⟹ s⊢y↝⇧+z ⟹
       (s⊢y↝⇧+y2 ⟹
        Inv_Space_Father_Is_Unique s ⟹ Inv_Space_HasNot_Loop s ⟹ z = Real r1 ⟹ y2 = Real r2 ⟹ r1 = r2) ⟹
       Inv_Space_Father_Is_Unique s ⟹ Inv_Space_HasNot_Loop s ⟹ z = Real r1 ⟹ y2 = Real r2 ⟹ r1 = r2"
  apply(induct rule:tran_path.induct)
  subgoal
    unfolding Inv_Space_Father_Is_Unique_def
     by (metis SonIsNoReal tran_path.cases)
   unfolding Inv_Space_Father_Is_Unique_def
   by metis

lemma YYIsUnique_Real:"(s ⊢ x ↝⇧+ y1) ⟹ (s ⊢ x ↝⇧+ y2) ⟹
Inv_Space_Father_Is_Unique s ⟹ Inv_Space_HasNot_Loop s ⟹ y1 = Real r1 ⟹ y2 = Real r2 ⟹
 r1 = r2"
  apply(induct rule:tran_path.induct)
  using tran_one_unique' apply simp
  using tran_more_unique' by blast

lemma "Inv_Space_Father_Is_Unique s ⟹ Inv_Space_HasNot_Loop s ⟹ Inv_Space_Path_Is_Unique s"
  unfolding Inv_Space_Path_Is_Unique_def
  using YYIsUnique_Real
  by auto

definition Inv_Space_Perms_IsNot_Empty::"State ⇒ bool" where
"Inv_Space_Perms_IsNot_Empty s ≡
            (∀sp v_page. s ⊢ (Virtual sp v_page) ⟶ get_perms s sp v_page ≠ {})"

definition Inv_Space_Spaces_In_Config::"State ⇒ bool" where
"Inv_Space_Spaces_In_Config s ≡ (spaces s ⊆ Address_Space SysConf)"

definition Inv_Space_InitialSpaces_In_Spaces::"State ⇒ bool" where
"Inv_Space_InitialSpaces_In_Spaces s ≡ (initialised_spaces s ⊆ spaces s)"

definition Inv_Space_Spaces_IsNot_None::"State ⇒ bool" where
"Inv_Space_Spaces_IsNot_None s ≡
    (spaces s = dom (space_threads s))"

definition Inv_NSpace_Is_None::"State ⇒ bool" where
"Inv_NSpace_Is_None s ≡
∀sp. sp ∈ Address_Space SysConf - spaces s ⟶ space_threads s sp = None"

lemma Inv_NSpace_Is_None':"Inv_Space_Spaces_IsNot_None s ⟹ Inv_NSpace_Is_None s ⟹
∀sp. sp ∈ Address_Space SysConf - spaces s ⟶ space_mapping s sp = None"
  unfolding Inv_Space_Spaces_IsNot_None_def Inv_NSpace_Is_None_def
    spaces_def dom_def by blast

definition Inv_Space_Vpage_Range::"State ⇒ bool" where
"Inv_Space_Vpage_Range s ≡ ∀sp v. s⊢(Virtual sp v) ⟶ v ≥ 0 ∧ v < page_maxnum"

definition Inv_Space_Sigma0Space_Vpage::"State ⇒ bool" where
"Inv_Space_Sigma0Space_Vpage s ≡ ∀v < page_maxnum. s⊢(Virtual Sigma0Space v) ↝⇧1 (Real v)"


definition Inv_Space::"State ⇒ bool" where
"Inv_Space s ≡ (Inv_Space_HasNot_Loop s ∧ Inv_Space_Valid_Has_Real s ∧
                 Inv_Space_Perms_IsNot_Empty s ∧ Inv_Space_Spaces_In_Config s ∧
                 Inv_Space_InitialSpaces_In_Spaces s ∧ Inv_Space_Perms_Subset s ∧
                 Inv_Space_Spaces_IsNot_None s ∧ Inv_Space_Vpage_Range s)"

definition Inv_Space_base::"State ⇒ bool" where
"Inv_Space_base s ≡ (spaces s ⊆ Address_Space SysConf) ∧
                    (initialised_spaces s ⊆ spaces s) ∧
                    (∀sp. sp ∈ Address_Space SysConf - spaces s ⟶
                          space_mapping s sp = None ∧
                          space_threads s sp = None) ∧
                    (∀sp. sp ∈ spaces s ⟶
                          space_mapping s sp ≠ None ∧
                          space_threads s sp ≠ None) ∧
                    (∀v. (s ⊢ (Virtual Sigma0Space v) ↝⇧1 (Real v)))"

section‹Thread›

definition Inv_Idle_NotIn_Threads::"State ⇒ bool" where
"Inv_Idle_NotIn_Threads s ≡ (idle ∉ threads s)"

definition Inv_Idle_Space_Is_KernelSpace::"State ⇒ bool" where
"Inv_Idle_Space_Is_KernelSpace s ≡ (thread_space s idle = Some KernelSpace)"

definition Inv_Sigma0_In_Active::"State ⇒ bool" where
"Inv_Sigma0_In_Active s ≡ (kSigma0 ∈ active_threads s)"

definition Inv_Sigma0_Space::"State ⇒ bool" where
"Inv_Sigma0_Space s ≡ (thread_space s kSigma0 = Some Sigma0Space)"

definition Inv_RootServer_In_Active::"State ⇒ bool" where
"Inv_RootServer_In_Active s ≡ (kRootServer ∈ active_threads s)"

definition Inv_RootServer_Space::"State ⇒ bool" where
"Inv_RootServer_Space s ≡ (thread_space s kRootServer = Some RootServerSpace)"

definition Inv_IntThreads_In_Active::"State ⇒ bool" where
"Inv_IntThreads_In_Active s ≡ (kIntThreads ⊆ active_threads s)"

definition Inv_IntThreads_Space::"State ⇒ bool" where
"Inv_IntThreads_Space s ≡ 
                 (∀t. t ∈ kIntThreads ⟶ thread_space s t = Some KernelSpace)"

definition Inv_Threads_In_Config::"State ⇒ bool" where
"Inv_Threads_In_Config s ≡ (threads s ⊆ Threads_Gno SysConf)"

definition Inv_Active_In_Threads::"State ⇒ bool" where
"Inv_Active_In_Threads s ≡ (active_threads s ⊆ threads s)"

definition Inv_NThreads_Is_None::"State ⇒ bool" where
"Inv_NThreads_Is_None s ≡
    (∀t. t ∈ Threads_Gno SysConf - threads s ⟶
          thread_space s t = None ∧ 
          thread_scheduler s t = None ∧
          thread_state s t = None ∧
          thread_pager s t = None ∧
          thread_handler s t = None ∧
          thread_message s t = None ∧
          thread_rcvWindow s t = None ∧
          thread_error s t = None ∧
          thread_priority s t = None ∧
          thread_total_quantum s t = None ∧
          thread_timeslice_length s t = None ∧
          thread_current_timeslice s t = None ∧
          thread_ipc_partner s t = None ∧
          thread_ipc_timeout s t = None ∧
          thread_incoming s t = None)"


definition Inv_Threads_IsNot_None::"State ⇒ bool" where
"Inv_Threads_IsNot_None s ≡
   (∀t. t ∈ threads s ⟶
          thread_state s t ≠ None ∧
          thread_priority s t ≠ None ∧
          thread_total_quantum s t ≠ None ∧
          thread_timeslice_length s t ≠ None ∧
          thread_current_timeslice s t ≠ None ∧
          thread_ipc_timeout s t ≠ None)"

definition Inv_Threads_Space_Dom::"State ⇒ bool" where
"Inv_Threads_Space_Dom s ≡ threads s = dom (thread_space s) - {idle}"

definition Inv_Threads_Space_In_Spaces::"State ⇒ bool" where
"Inv_Threads_Space_In_Spaces s ≡
          (∀t. t ∈ threads s ⟶ (∃space. space ∈ spaces s ∧ thread_space s t = Some space))"

definition Inv_Threads_Eq_Space_Threads::"State ⇒ bool" where
"Inv_Threads_Eq_Space_Threads s ≡
∀space. space ∈ spaces s ⟶ the (space_threads s space) = {t. thread_space s t = Some space}"


lemma Thread_Exist:"Inv_Threads_Space_In_Spaces s ⟹ Inv_Threads_Space_Dom s ⟹ space ∉ spaces s ⟹ 
∄t. t ≠ idle ∧ thread_space s t = Some space"
unfolding Inv_Threads_Space_In_Spaces_def Inv_Threads_Space_Dom_def
by auto

lemma Thread_Exist':"⋀t. Inv_Threads_Space_In_Spaces s ⟹ Inv_Threads_Space_Dom s ⟹ space ∉ spaces s ⟹ 
t ≠ idle ⟹ thread_space s t = Some space ⟹ False"
unfolding Inv_Threads_Space_In_Spaces_def Inv_Threads_Space_Dom_def
by auto

lemma Thread_AtLeast_One_In_Space:
  assumes a1:"Inv_Threads_Space_In_Spaces s"
      and a2:"Inv_Threads_Eq_Space_Threads s"
      and a3:"gno ∈ threads s"
    shows "{gno} ⊆ the (space_threads s (the (thread_space s gno)))"
proof-
  from a3 obtain space where h1:"space ∈ spaces s ∧ thread_space s gno = Some space" 
    using a1[unfolded Inv_Threads_Space_In_Spaces_def] by auto
  then have h12:"the (space_threads s space) = {t. thread_space s t = Some space}"
    using a2[unfolded Inv_Threads_Eq_Space_Threads_def] by simp
  then show "{gno} ⊆ the (space_threads s (the (thread_space s gno)))"
    using h1 by simp
qed

definition Inv_Threads_Sche_In_Threads::"State ⇒ bool" where
"Inv_Threads_Sche_In_Threads s ≡
          (∀t. t ∈ threads s ⟶ (∃sche. sche ∈ threads s ∧ thread_scheduler s t = Some sche))"

definition Inv_Threads_Partner_In_Threads::"State ⇒ bool" where
"Inv_Threads_Partner_In_Threads s ≡
          (∀t. t ∈ threads s ⟶ 
                    (∃partner. (thread_ipc_partner s t = Some partner) ∧ 
                    (partner ≠ NILTHREAD ∧ partner ≠ ANYTHREAD ⟶ 
                              TidToGno partner ∈ threads s)))"

definition Inv_Threads_Incoming_In_Threads::"State ⇒ bool" where
"Inv_Threads_Incoming_In_Threads s ≡
          (∀t. t ∈ threads s ⟶ 
          (∃send_l. (thread_incoming s t = Some send_l) ∧ (∀gno. gno ∈ set (send_l) ⟶ gno ∈ threads s)))"

definition Inv_NActive_Utcb_Is_None::"State ⇒ bool" where
"Inv_NActive_Utcb_Is_None s ≡
 (∀t. t ∈ threads s ⟶
     (t ∉ active_threads s ⟶
           thread_pager s t = None ∧
           thread_handler s t = None ∧
           thread_message s t = None ∧
           thread_rcvWindow s t = None ∧
           thread_error s t = None))"

definition Inv_Active_Utcb_IsNot_None::"State ⇒ bool" where
"Inv_Active_Utcb_IsNot_None s ≡
 (∀t. t ∈ active_threads s ⟶
     (t ∉ kIntThreads ⟶
           thread_pager s t ≠ None ∧
           thread_handler s t ≠ None ∧
           thread_message s t ≠ None ∧
           thread_rcvWindow s t ≠ None ∧
           thread_error s t ≠ None))"

definition Inv_IntThreads_Utcb_Is_None::"State ⇒ bool" where
"Inv_IntThreads_Utcb_Is_None s ≡
(∀t. t ∈ active_threads s ⟶
    (t ∈ kIntThreads ⟶
           thread_pager s t = None ∧
           thread_handler s t = None ∧
           thread_message s t = None ∧
           thread_rcvWindow s t = None ∧
           thread_error s t = None))"

definition Inv_Thread::"State ⇒ bool" where
"Inv_Thread s ≡ (Inv_Idle_NotIn_Threads s ∧ Inv_Idle_Space_Is_KernelSpace s ∧
    Inv_Sigma0_In_Active s ∧ Inv_Sigma0_Space s ∧ Inv_IntThreads_Space s ∧
    Inv_RootServer_In_Active s ∧ Inv_RootServer_Space s ∧ Inv_IntThreads_In_Active s ∧ 
    Inv_Threads_In_Config s ∧ Inv_Active_In_Threads s ∧ Inv_NThreads_Is_None s ∧ Inv_Threads_Space_Dom s ∧
    Inv_Threads_IsNot_None s ∧ Inv_Threads_Space_In_Spaces s ∧ Inv_Threads_Eq_Space_Threads s ∧ 
    Inv_Threads_Sche_In_Threads s ∧ Inv_NActive_Utcb_Is_None s ∧ Inv_Active_Utcb_IsNot_None s ∧ 
    Inv_IntThreads_Utcb_Is_None s ∧ Inv_Threads_Partner_In_Threads s ∧ Inv_Threads_Incoming_In_Threads s)"

definition Invariants::"State ⇒ bool" where
"Invariants s ≡ Inv_Current s ∧ Inv_Space s ∧ Inv_Thread s"
