theory L4_Schedule
imports L4_Thread
begin

definition TIMER_TICK_LENGTH:: int where
"TIMER_TICK_LENGTH ≡ 2000"

definition TOTAL_QUANTUM_EXPIRED :: int where
"TOTAL_QUANTUM_EXPIRED ≡ max_int"

definition SetCurrentTime::"State ⇒ int ⇒ State" where
"SetCurrentTime S cur_time = S⦇current_time:=cur_time⦈"

definition GetCurrentTime::"State ⇒ int" where
"GetCurrentTime S = current_time S"

definition SetCurrentTimeslice::"State ⇒ int ⇒ State" where
"SetCurrentTimeslice S time = S⦇current_timeslice:= time⦈"

definition GetCurrentTimeslice::"State ⇒ int" where
"GetCurrentTimeslice S = current_timeslice S"

fun find_max_queuing_thread :: "(prio ⇒ globalid_t list) ⇒ prio ⇒ globalid_t" where
"find_max_queuing_thread queuing prio =
 (if prio = 0
  then (if queuing 0 ≠ [] then last (queuing 0) else idle)
  else (if queuing prio ≠ [] then last (queuing prio) else find_max_queuing_thread queuing (prio-1)))"

definition find_next_thread :: "State ⇒ globalid_t" where
"find_next_thread S = find_max_queuing_thread (ready_queuing S) 255"

definition switch_to_idle::"State ⇒ State" where
"switch_to_idle S = S⦇current_thread := idle⦈"

definition switch_to::"State ⇒ globalid_t ⇒ State" where
"switch_to S to = S⦇current_thread := to⦈"

definition Scheduling :: "State ⇒ globalid_t ⇒ State" where
"Scheduling S current ≡ 
(let gno = find_next_thread S
  in if gno ≠ current 
     then (if current ≠ idle
           then switch_to (enqueue_ready S current) gno
           else switch_to S gno)
    else S)"

fun find_expired_threads::"State ⇒ globalid_t list ⇒ int ⇒ globalid_t list" where
"find_expired_threads S [] cur_time = []"|
"find_expired_threads S (t#ts) cur_time =
 (if GetTimeout(GetIpcTimeout S t) ≤ cur_time
  then t#find_expired_threads S ts cur_time
  else find_expired_threads S ts cur_time)"

fun find_highest_thread::"State ⇒ globalid_t list ⇒ globalid_t ⇒ globalid_t" where
"find_highest_thread S [] current = current" |
"find_highest_thread S (t#ts) current =
 (if GetThreadPriority S t > GetThreadPriority S current
  then find_highest_thread S ts t
  else find_highest_thread S ts current)"

fun modify_expired_threads_state::"State ⇒ globalid_t list ⇒ State" where
"modify_expired_threads_state s [] = s" |
"modify_expired_threads_state s (x#xs) = 
      SetThreadState (modify_expired_threads_state s xs) x tsRunning"

definition parse_wakeup_queues::"State ⇒ globalid_t ⇒ State × globalid_t" where
"parse_wakeup_queues S current =
(let cur_time = current_time S;
     wait_list = wait_queuing S;
     expired_list = find_expired_threads S wait_list cur_time;
     highest_thread = find_highest_thread S expired_list current;
     ready_list = ready_queuing S;
     S1 = dequeue_waits S expired_list;
     S2 = enqueue_readys S1 expired_list;
     S3 = modify_expired_threads_state S2 expired_list
 in (S3, highest_thread))"

definition end_of_timeslice::"State ⇒ globalid_t ⇒ State" where
"end_of_timeslice S gno =
(let S1 = enqueue_ready S gno;
     gno_queue = ready_queuing S1 (GetThreadPriority S1 gno);
     next = last gno_queue;
     S2 = SetThreadCurrentTimeslice S1 gno (current_timeslice S1 + GetThreadCurrentTimeslice S1 gno);
     S3 = S2⦇current_timeslice:= GetThreadCurrentTimeslice S2 next⦈
 in S3)"

definition preempt_thread::"State ⇒ globalid_t ⇒ globalid_t ⇒ State" where
"preempt_thread S current dest = 
(let S1 = SetThreadCurrentTimeslice S current (GetCurrentTimeslice S);
     S2 = SetCurrentTimeslice S1 (GetThreadCurrentTimeslice S1 dest);
     S3 = enqueue_ready S2 current
 in S3)"

definition total_quantum_expired::"State ⇒ globalid_t ⇒ State" where
"total_quantum_expired S current = S"

definition update_timeslice::"State ⇒ State" where
"update_timeslice S =
(let current = current_thread S;
     total_quantum = GetThreadQuantum S current;
     timeslice_length = GetThreadTimeslice S current;
     current_timeslice = GetThreadCurrentTimeslice S current
 in (if total_quantum ≠ 0
     then (if total_quantum = TOTAL_QUANTUM_EXPIRED
              then total_quantum_expired S current
              else 
           if total_quantum ≤ timeslice_length
              then SetThreadQuantum (SetThreadCurrentTimeslice S current (current_timeslice + total_quantum)) current TOTAL_QUANTUM_EXPIRED
              else SetThreadQuantum S current (total_quantum - timeslice_length))
     else S))"

definition handle_time_interrupt::"State ⇒ State" where
"handle_time_interrupt S =
(let current_thread = current_thread S;
     S1 = SetCurrentTime S (current_time S + TIMER_TICK_LENGTH);
     S2 = fst(parse_wakeup_queues S1 current_thread);
     wakeup = snd(parse_wakeup_queues S1 current_thread);
     S6 = (if current_thread = idle then S2 else let
     cur_timeslice = (current_timeslice S2 - TIMER_TICK_LENGTH);
     S3 = S2⦇current_timeslice:=cur_timeslice⦈;
     S4 = end_of_timeslice S3 current_thread;
     S5 = (if ¬(cur_timeslice ≤ 0 ∧ thread_timeslice_length S2 current_thread ~= Some 0)
           then (if wakeup = current_thread then S3
                 else switch_to (preempt_thread S3 current_thread wakeup) wakeup)
           else (Scheduling ((update_timeslice S4)) current_thread)) in S5)
in S6)"

definition "Timer = handle_time_interrupt"
end
