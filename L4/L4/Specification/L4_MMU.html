theory L4_MMU
  imports L4_Thread
begin

section‹ MMU and Lookup ›
subsection‹ Page Table ›
definition next_page::"State ⇒ spaceName_t ⇒ v_page_t ⇒ page_t option" where
"next_page s sp v_page =
(if space_mapping s sp = None
 then None
 else (let mapping = the (space_mapping s sp)
        in (if mapping v_page = None
            then None
            else Some (fst (the (mapping v_page))))))"

definition next_space::"State ⇒ spaceName_t ⇒ v_page_t ⇒ spaceName_t option" where
"next_space s sp v_page =
(if space_mapping s sp = None
 then None
 else (let mapping = the (space_mapping s sp)
        in (if mapping v_page = None
            then None
            else (let page = fst (the (mapping v_page))
                   in (case page of Virtual sp' v_page' ⇒ Some sp' |
                                    Real _ ⇒ None)))))"

definition next_vpage::"State ⇒ spaceName_t ⇒ v_page_t ⇒ v_page_t option" where
"next_vpage s sp v_page =
(if space_mapping s sp = None
 then None
 else (let mapping = the (space_mapping s sp)
        in (if mapping v_page = None
            then None
            else (let page = fst (the (mapping v_page))
                   in case page of Virtual sp' v_page' ⇒ Some v_page' |
                                   Real _ ⇒ None))))"

definition pt_walk::"State ⇒ spaceName_t ⇒ v_page_t ⇀ r_page_t × perms_t set" where
"pt_walk s sp v_page ≡ (if (s ⊢ (Virtual sp v_page))
                         then Some ((SOME r_page. (s ⊢ (Virtual sp v_page) ↝⇧+ (Real r_page))), get_perms s sp v_page)
                         else None)"

subsection‹ TLB ›

definition entry_range::"tlb_entry ⇒ vaddr_t set" where
"entry_range e = (case e of PTE _ v _ _ ⇒
{addr. addr ≥ vpage_to_vaddr v ∧ addr ≤ vpage_to_vaddr v + page_size - 1})"

definition asid_of::"tlb_entry ⇒ asid" where
"asid_of e = (case e of PTE a _ _ _ ⇒ a)"

definition entry_set::"tlb_entry set ⇒ asid ⇒ vaddr_t ⇒ tlb_entry set" where
"entry_set t a vaddr = {e. e ∈ t ∧ (vaddr ∈ entry_range e) ∧ a = asid_of e}"

definition selective_invalidation::"tlb_entry set ⇒ asid ⇒ vaddr_t ⇒ tlb_entry set" where
"selective_invalidation t a vaddr = t - entry_set t a vaddr"

definition asid_invalidation::"tlb_entry set ⇒ asid ⇒ tlb_entry set" where
"asid_invalidation t a = t - {e. e ∈ t ∧ asid_of e = a}"

definition vaddr_invalidation::"tlb_entry set ⇒ vaddr_t ⇒ tlb_entry set" where
"vaddr_invalidation t vaddr = t - {e. e ∈ t ∧ vaddr ∈ entry_range e}"

definition update_tlb_select::"State ⇒ asid ⇒ vaddr_t ⇒ State" where
"update_tlb_select s a vaddr = s⦇tlb:= (selective_invalidation (tlb s) a vaddr)⦈"

definition update_tlb_asid::"State ⇒ asid ⇒ State" where
"update_tlb_asid s a = s⦇tlb:= (asid_invalidation (tlb s) a)⦈"

definition update_tlb_vaddr::"State ⇒ vaddr_t ⇒ State" where
"update_tlb_vaddr s vaddr = s⦇tlb:= (vaddr_invalidation (tlb s) vaddr)⦈"

definition lookup_tlb::"State ⇒ asid ⇒ vaddr_t ⇒ lookup_type" where
"lookup_tlb s a vaddr =
(let S = entry_set (tlb s) a vaddr
  in (if S = {} then Miss else
      if (∃e. S = {e}) then Hit (the_elem S) else Incon))"

lemma lookup_tlb_asid:"lookup_tlb s a vaddr = Hit entry ⟶ a = asid_of entry"
  unfolding lookup_tlb_def entry_set_def Let_def the_elem_def
  by auto

definition vaddr_to_paddr::"vaddr_t ⇒ tlb_entry ⇒ paddr_t" where
"vaddr_to_paddr vaddr entry =
(case entry of PTE a v_page r_page _ ⇒
 rpage_to_paddr r_page + vaddr - vpage_to_vaddr v_page)"

definition mmu_translate::"State ⇒ vaddr_t ⇒ State × (paddr_t option)" where
"mmu_translate s vaddr = 
(let
 a = GetCurrentSpace s;
 result = lookup_tlb s a vaddr    ―‹ Walk TLB ›
 in (case result of
 Miss ⇒ (let v_page = vaddr_to_vpage vaddr;
              pt_r = pt_walk s a v_page ―‹ Walk Page Table ›
           in (if pt_r = None 
               then (s, None) ―‹ Page Fault ›
               else
                 let perms = snd (the pt_r);
                     r_page = fst (the pt_r);
                     entry = PTE a v_page r_page perms
                  in (s⦇tlb:= tlb s ∪ {entry}⦈,Some (vaddr_to_paddr vaddr entry))))|
 Hit entry ⇒ (s,Some (vaddr_to_paddr vaddr entry))|
 Incon ⇒ (s,None)))" ―‹ Inconsistency ›

definition mem_read::"State ⇒ paddr_t ⇒ byte" where
"mem_read s paddr = heap s paddr"

definition mem_write::"State ⇒ paddr_t ⇒ byte ⇒ State" where
"mem_write s paddr val = s⦇heap:= (heap s)(paddr := val)⦈"

definition mmu_read::"State ⇒ vaddr_t ⇀ byte" where
"mmu_read s vaddr =
(let space = GetCurrentSpace s;
     vpage = vaddr_to_vpage vaddr
 in (if pfRead ∉ get_perms s space vpage
     then (let paddr = snd (mmu_translate s vaddr)
           in (if paddr = None ―‹ Exception ›
               then None ―‹ Reading Failure ›
               else Some (mem_read s (the paddr))))
     else None))" ―‹ No Permission ›

definition mmu_write::"State ⇒ vaddr_t ⇒ byte ⇒ State × bool" where
"mmu_write s vaddr val =
(let space = GetCurrentSpace s;
     vpage = vaddr_to_vpage vaddr
 in (if pfWrite ∉ get_perms s space vpage
     then (let paddr = snd (mmu_translate s vaddr)
           in (if paddr = None ―‹ Exception ›
               then (s,False) ―‹ Writing Failure ›
               else (mem_write s (the paddr) val, True)))
     else (s,False)))" ―‹ No Permission ›
end
