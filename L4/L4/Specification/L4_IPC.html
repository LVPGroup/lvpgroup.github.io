theory L4_IPC
imports L4_Schedule
begin

definition canSend :: "State ⇒ globalid_t ⇒ bool" where
"canSend s t = (GetThreadState s t ∈ {tsRunning, tsPolling})"

definition canReceive :: "State ⇒ globalid_t ⇒ bool" where
"canReceive s t = (GetThreadState s t ∈ {tsWaitingForever, tsWaitingTimeout})"

definition copy_data :: "State ⇒ globalid_t ⇒ globalid_t ⇒ State" where
"copy_data S from to =
  (if thread_message S from ≠ None
   then SetThreadMessage S to (GetThreadMessage S from)
   else S)"

definition PerformIPC :: "State ⇒ globalid_t ⇒ globalid_t ⇒ State" where
"PerformIPC S from to =
(if from ∈ active_threads S ∧ from ∉ kIntThreads ∧ to ∈ active_threads S ∧ to ∉ kIntThreads ∧ canSend S from ∧ canReceive S to ∧
    thread_message S from ≠ None
 then
    let  from_space = GetThreadSpace S from;
         to_space = GetThreadSpace S to;
         from_msg = (GetThreadMessage S from);
         S1 = copy_data S from to
     in (case from_msg of UNTYPED n ⇒ S1 | STRING str ⇒ S1 | 
        MAP f_fpage ⇒ if fpage_type f_fpage = valid_fpage then
         let t_fpage = (GetThreadRcvWindow S to);
             f_base = fp_base f_fpage;
             f_size = fp_size f_fpage;
             f_perms = fp_perms f_fpage;
             t_base = fp_base t_fpage
          in Map_fpage S1 from_space f_base to_space t_base f_perms f_size
        else S |
        GRANT f_fpage ⇒ if fpage_type f_fpage = valid_fpage then
         let t_fpage = (GetThreadRcvWindow S to);
             f_base = fp_base f_fpage;
             f_size = fp_size f_fpage;
             f_perms = fp_perms f_fpage;
             t_base = fp_base t_fpage
          in Grant_fpage S1 from_space f_base to_space t_base f_perms f_size
        else S)
 else S)"

definition canIPC :: "State ⇒ globalid_t ⇒ bool" where
"canIPC s = (λt. t ∈ active_threads s ∧ 
(t ∈ kIntThreads ⟶ (GetThreadState s t = tsHalted)) ∧ 
(t ∉ kIntThreads ⟶ (GetThreadState s t = tsRunning)))"

―‹only wait›
definition JustWait_Cond :: "Sys_Config ⇒ State ⇒ globalid_t ⇒ timeout_t ⇒ threadid_t ⇒ bool" where
"JustWait_Cond SysConf S gno timeout fromSpecifier ≡
  (canIPC S gno) ∧ ¬(isNoTimeout timeout) ∧ (fromSpecifier ≠ NILTHREAD) ∧
  (fromSpecifier ∈ (GetThreadsTids S ∪ {ANYTHREAD})) ∧
  (fromSpecifier = ANYTHREAD ⟶ thread_incoming S gno = Some []) ∧
  ((fromSpecifier ≠ ANYTHREAD) ⟶ TidToGno fromSpecifier ∉ set (GetThreadsIncoming S gno))"

―‹(find (λt. t = fromSpecifier) (GetThreadsIncomingTids SysConf S gno)) = None›
definition JustWait :: "Sys_Config ⇒ State ⇒ globalid_t ⇒ timeout_t ⇒ threadid_t ⇒ State" where
"JustWait SysConf S gno timeout fromSpecifier = 
(if JustWait_Cond SysConf S gno timeout fromSpecifier 
 then (let S1 = SetIpcPartner S gno fromSpecifier
        in (if isInfinite timeout
            then SetThreadState S1 gno tsWaitingForever
            else enqueue_wait (SetIpcTimeout (SetThreadState S1 gno tsWaitingTimeout) gno timeout) gno))
 else S)"

―‹send  fail ;  send and receive  fail ›
definition SetUpReceivePhaseAndPoll_Cond::"Sys_Config ⇒ State ⇒ globalid_t ⇒ globalid_t ⇒ timeout_t ⇒ timeout_t ⇒ threadid_t ⇒ bool" where
"SetUpReceivePhaseAndPoll_Cond SysConf S tcb_from tcb_to poll_timeout recv_timeout fromSpecifier ≡
  (canIPC S tcb_from) ∧ (tcb_to ∈ threads S) ∧
  ((tcb_to ∈ dom (thread_ipc_partner S) ∧ isWaiting (GetThreadState S tcb_to)) ⟶
            ((thread_ipc_partner S tcb_to) ≠ Some (GnoToTid tcb_from)) ∧
            ((thread_ipc_partner S tcb_to) ≠ Some ANYTHREAD)) ∧
  (fromSpecifier ∈ GLOBAL_TID SysConf) ∧ ¬(isNoTimeout poll_timeout) ∧
  (fromSpecifier ∈ (GetThreadsTids S ∪ {ANYTHREAD,NILTHREAD}))"

definition SetUpReceivePhaseAndPoll::"Sys_Config ⇒ State ⇒ globalid_t ⇒ globalid_t ⇒ timeout_t ⇒ timeout_t ⇒ threadid_t ⇒ State" where
"SetUpReceivePhaseAndPoll SysConf S tcb_from tcb_to poll_timeout recv_timeout fromSpecifier = 
(if SetUpReceivePhaseAndPoll_Cond SysConf S tcb_from tcb_to poll_timeout recv_timeout fromSpecifier
 then (let S1 = SetIpcPartner S tcb_from (GnoToTid tcb_to);
           S2 = SetThreadsIncoming S1 tcb_to (GetThreadsIncoming S1 tcb_to @ [tcb_from]);
           S3 = SetThreadState S2 tcb_from tsPolling;
           S4 = (if isInfinite poll_timeout then S3 else enqueue_wait (SetIpcTimeout S3 tcb_from poll_timeout) tcb_from)
        in (if fromSpecifier ≠ NILTHREAD
            then (if isNoTimeout recv_timeout 
                  then SetError S4 tcb_from eRecvTimeout 
                  else(let S5 = SetRecvFor S4 tcb_from fromSpecifier
                   in SetRecvTimeout S5 tcb_from recv_timeout))
            else S4))
 else S)"

―‹only receive›
definition JustReceive_Cond::"Sys_Config ⇒ State ⇒ threadid_t ⇒ bool" where
"JustReceive_Cond SysConf S fromSpecifier ≡
  (canIPC S (current_thread S)) ∧ (fromSpecifier ∈ GLOBAL_TID SysConf) ∧ (fromSpecifier ≠ NILTHREAD) ∧
  ((fromSpecifier ≠ ANYTHREAD) ⟶ (find (λt. t = fromSpecifier) ((GetThreadsIncomingTids SysConf S (current_thread S)))) ≠ None) ∧
  ((fromSpecifier = ANYTHREAD) ⟶ (thread_incoming S (current_thread S) ≠ Some []))"

definition JustReceive::"Sys_Config ⇒ State ⇒ threadid_t ⇒ State" where
"JustReceive SysConf S fromSpecifier = 
(if JustReceive_Cond SysConf S fromSpecifier
 then (let tcb_from =
              if (fromSpecifier = ANYTHREAD)
              then hd (the (thread_incoming S (current_thread S)))
              else TidToGno fromSpecifier;
           S1 = SetThreadsIncomingDel S (current_thread S) tcb_from
        in (if cond
            then (let S3 = PerformIPC S1 tcb_from (current_thread S1);
                      S4 = SetError S3 (current_thread S3) eNoError;
                      S5 = SetError S4 tcb_from eNoError
                   in (if tcb_from ∈ dom (thread_recv_for S5) ∧ tcb_from ∈ dom (thread_recv_timeout S5)                            ― ‹tcb_from also have a recv phase›
                       then (let rcv_timeout = (GetRecvTimeout S5 tcb_from);
                                 S6 = SetIpcPartner S5 tcb_from (GetRecvFor S5 tcb_from)
                              in (if isInfinite rcv_timeout
                                  then SetThreadState S6 tcb_from tsWaitingForever
                                   else(
                                        let S7 = SetThreadState S6 tcb_from tsWaitingTimeout
                                        in enqueue_wait (SetIpcTimeout S7 tcb_from rcv_timeout) tcb_from
                                       )))
                       else dequeue_wait (UnWait S5 tcb_from) tcb_from))
            else (let S3 = dequeue_wait(UnWait S1 tcb_from) tcb_from;
                      error = (SOME err. err ∈ dIpcFailures)
                   in SetError (SetError S3 (current_thread S3) error) tcb_from error)))
 else S)"

―‹send  success ;  send and receive  success ›
definition WakeDestThenWait_Cond::"Sys_Config ⇒ State ⇒ globalid_t ⇒ globalid_t ⇒ timeout_t ⇒ threadid_t ⇒ bool" where
"WakeDestThenWait_Cond SysConf S tcb_from tcb_to recv_timeout fromSpecifier ≡
  (canIPC S tcb_from) ∧ (tcb_to ∈ threads S) ∧ (isWaiting (GetThreadState S tcb_to)) ∧
  (tcb_to ∈ dom (thread_ipc_partner S)) ∧ (GetIpcPartner S tcb_to ∈ {GnoToTid tcb_from, ANYTHREAD}) ∧
  (fromSpecifier ∈ GLOBAL_TID SysConf) ∧
  (fromSpecifier ∈ (GetThreadsTids S ∪{ANYTHREAD, NILTHREAD}))"

definition WakeDestThenWait::"Sys_Config ⇒ State ⇒ globalid_t ⇒ globalid_t ⇒ timeout_t ⇒ threadid_t ⇒ State" where
"WakeDestThenWait SysConf S tcb_from tcb_to recv_timeout fromSpecifier = 
(if WakeDestThenWait_Cond SysConf S tcb_from tcb_to recv_timeout fromSpecifier
 then (if cond
       then (let S1 = PerformIPC S tcb_from tcb_to;
                 S2 = SetError S1 tcb_to eNoError;
                 S3 = SetError S2 tcb_from eNoError
              in (if fromSpecifier = NILTHREAD
                  then dequeue_wait (UnWait S3 tcb_to) tcb_to
                  else (if isNoTimeout recv_timeout 
                        then SetError S3 tcb_from eRecvTimeout 
                        else (if isInfinite recv_timeout
                              then dequeue_wait(WakeUpAndWait S3 tcb_from tcb_to tsWaitingForever) tcb_to
                              else dequeue_wait(enqueue_wait (WakeUpAndWait S3 tcb_from tcb_to tsWaitingTimeout) tcb_from) tcb_to))))
       else (let S1 = dequeue_wait(UnWait S tcb_to) tcb_to;
                 error = (SOME e. e ∈ dIpcFailures)
              in (SetError (SetError S1 tcb_to error) tcb_from error)))
 else S)"

end
