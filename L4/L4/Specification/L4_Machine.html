definition step::"State ⇒ Event ⇒ State" where
"step S e ≡
(case e of syscall (eIPC to_gno fromSpecifier send_timeout recv_timeout)
                  ⇒ IPC SysConf S to_gno fromSpecifier send_timeout recv_timeout |
           syscall (eThreadControl destNo spaceSpec schedNo pagerNo)
                  ⇒ ThreadControl SysConf S destNo spaceSpec schedNo pagerNo |
           syscall (eSpaceControl spaceSpec control KernelInterfacePageArea UtcbArea)
                  ⇒ SpaceControl SysConf S spaceSpec control KernelInterfacePageArea UtcbArea |
           syscall (eExchangeRegister gno control sp ip flags pager handle)
                  ⇒ ExchangeRegister SysConf S gno control sp ip flags pager handle |
           syscall (eThreadSwitch dest)
                  ⇒ ThreadSwitch SysConf S dest |
           syscall (eSchedule destNo timeControl procControl prio preemptControl)
                  ⇒ Schedule SysConf S destNo timeControl procControl prio preemptControl |
           syscall (eUnmap control msg)
                  ⇒ Unmap S control msg |
           syscall (eProcessorControl processor_no internal_frequency external_frequency voltage)
                  ⇒ ProcessorControl S processor_no internal_frequency external_frequency voltage |
           syscall (eMemoryControl attr1 attr2 attr3 attr4)
                  ⇒ MemoryControl S attr1 attr2 attr3 attr4 |
           syscall eSystemClock ⇒ fst(SystemClock S) |
           timer ⇒ Timer S |
           interrupt i ⇒ S)"

primrec run::"State ⇒ Event list ⇒ State" where
"run s [] = s" |
"run s (e#es) = run (step s e) es"

definition "Reachable s ≡ ∃es. run s0_req es = s"
