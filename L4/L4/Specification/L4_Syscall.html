definition IPC::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ timeout_t ⇒ timeout_t ⇒ State" where
"IPC SysConf S to_gno fromSpecifier send_timeout recv_timeout =
(if (current_thread S ∈ active_threads S) ∧ (current_thread S ∉ kIntThreads) ∧ (canIPC S (current_thread S)) ∧ (to_gno ∈ GLOBAL_TID SysConf) ∧ (fromSpecifier ∈ GLOBAL_TID SysConf)
 then (if to_gno = ANYTHREAD
       then SetError S (current_thread S) eSendNonExistingPartner
       else (if fromSpecifier ∉ (GetThreadsTids S ∪ {ANYTHREAD, NILTHREAD})
             then SetError S (current_thread S) eRecvNonExistingPartner
             else (if to_gno ≠ NILTHREAD ∧ (to_gno ∉ GetThreadsTids S)
                   then SetError S (current_thread S) eSendNonExistingPartner
                   else WeakIPC SysConf S to_gno fromSpecifier send_timeout recv_timeout)))
 else S)"

definition ThreadControl_Cond::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ bool" where
"ThreadControl_Cond SysConf S destNo spaceSpec schedNo pagerNo ≡
    (current_thread S ∈ active_threads S) ∧ (current_thread S ∉ kIntThreads) ∧
    (thread_state S (current_thread S) = Some tsRunning) ∧ 
    (destNo ∈ GLOBAL_TID SysConf) ∧ (spaceSpec ∈ GLOBAL_TID SysConf) ∧
    (schedNo ∈ GLOBAL_TID SysConf) ∧ (pagerNo ∈ GLOBAL_TID SysConf)"

definition ThreadControl_Delete_Cond::"State ⇒ threadid_t ⇒ bool" where
"ThreadControl_Delete_Cond S destNo ≡
  (destNo ∈ GetThreadsTids S) ∧ (destNo ≠ GnoToTid (current_thread S)) ∧
  (thread_space S (TidToGno destNo) ≠ Some Sigma0Space) ∧
  (thread_space S (TidToGno destNo) ≠ Some RootServerSpace) ∧
  (thread_space S (TidToGno destNo) ≠ Some KernelSpace)"

definition ThreadControl_Create_Cond::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ bool" where
"ThreadControl_Create_Cond SysConf S destNo spaceSpec schedNo pagerNo ≡
    (destNo ∈ GLOBAL_GNO SysConf) ∧ (spaceSpec ∈ GLOBAL_GNO SysConf) ∧ (schedNo ∈ GetThreadsTids S) ∧
    (pagerNo ∈ (GetThreadsTids S ∪ {NILTHREAD})) ∧ (threads S ≠ Threads_Gno SysConf) ∧
    (spaceSpec = destNo ⟶ spaces S ≠ Address_Space SysConf) ∧
    (spaceSpec ≠ destNo ⟶ (spaceSpec ∈ GetThreadsTids S) ∧ GetThreadSpace S (TidToGno spaceSpec) ∈ spaces S) ∧
    ((pagerNo ∈ GetThreadsTids S) ⟶  (spaceSpec ∈ GetThreadsTids S) ∧ (GetThreadSpace S (TidToGno spaceSpec) ∈ initialised_spaces S) ∧ 
           (TidToGno schedNo ∈ active_threads S)) ∧
    ((spaceSpec ∈ GetThreadsTids S) ⟶ 
                (GetSpaceThreadsNums S (GetThreadSpace S (TidToGno spaceSpec)) < MaxThreadsPerSpace SysConf))"

definition ThreadControl_Modify_Cond::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ bool" where
"ThreadControl_Modify_Cond SysConf S destNo spaceSpec schedNo pagerNo ≡
  (destNo ∈ GetThreadsTids S) ∧ (spaceSpec ∈ GetThreadsTids S) ∧
  (schedNo ∈ GetThreadsTids S ∪ {NILTHREAD}) ∧ (pagerNo ∈ GetThreadsTids S ∪ {NILTHREAD}) ∧
  ((pagerNo ≠ NILTHREAD) ⟶ GetThreadSpace S (TidToGno spaceSpec) ∈ initialised_spaces S) ∧
  ((pagerNo ≠ NILTHREAD) ∧ (destNo ∉ GetActiveThreadsTids S) ⟶ 
      ((schedNo ≠ NILTHREAD) ⟶ (TidToGno schedNo ∈ active_threads S)) ∧ 
      ((schedNo = NILTHREAD) ⟶ (GetThreadScheduler S (TidToGno destNo) ∈ active_threads S))) ∧
  (GetThreadSpace S (TidToGno spaceSpec) ≠ GetThreadSpace S (TidToGno destNo) ⟶
      GetSpaceThreadsNums S (GetThreadSpace S (TidToGno spaceSpec)) < MaxThreadsPerSpace SysConf)"

definition ThreadControl::"Sys_Config ⇒ State ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ threadid_t ⇒ State" where
"ThreadControl SysConf S destNo spaceSpec schedNo pagerNo =
(if ThreadControl_Cond SysConf S destNo spaceSpec schedNo pagerNo
 then (if ¬ dIsPrivilegedSpace ( GetCurrentSpace S)
       then SetError S (current_thread S) eNoPrivilege
       else (if spaceSpec = NILTHREAD 
               then (if ThreadControl_Delete_Cond S destNo
                     then WeakDeleteThread SysConf S destNo
                     else SetError S (current_thread S) (SOME e. e∈{eUnavailableThread,eNoPrivilege}))
               else
            (if (spaceSpec ≠ NILTHREAD) ∧ (destNo ∉ GetThreadsTids S)
               then (if ThreadControl_Create_Cond SysConf S destNo spaceSpec schedNo pagerNo
                     then WeakCreateThread SysConf S destNo spaceSpec schedNo pagerNo
                     else SetError S (current_thread S) (SOME e. e ∈{eInvalidSpace,eUnavailableThread,eInvalidScheduler,
                              eUnavailableThread,eOutOfMemory}))
               else
            (if (spaceSpec ≠ NILTHREAD) ∧ (destNo ∈ GetThreadsTids S) ∧ (TidToGno destNo ∉ kIntThreads)
               then (if ThreadControl_Modify_Cond SysConf S destNo spaceSpec schedNo pagerNo
                     then WeakModifyThread SysConf S destNo spaceSpec schedNo pagerNo
                     else SetError S (current_thread S) (SOME e. e ∈{eInvalidSpace,eUnavailableThread,eInvalidScheduler,eOutOfMemory}))
               else
            (if (spaceSpec ≠ NILTHREAD) ∧ (destNo ∈ GetThreadsTids S) ∧ (TidToGno destNo ∈ kIntThreads)
               then (if pagerNo ∈ GetThreadsTids S 
                      then IntThreadControl SysConf S destNo pagerNo
                      else SetError S (current_thread S) eUnavailableThread)
               else S)))))
 else S)"

definition SpaceControl::"Sys_Config ⇒ State ⇒ globalid_t ⇒ nat ⇒ fpage_t ⇒ fpage_t ⇒ State" where
"SpaceControl SysConf S spaceSpec control KernelInterfacePageArea UtcbArea = 
(if (current_thread S ∈ threads S) ∧
    (current_thread S ∈ active_threads S) ∧
    (current_thread S ∉ kIntThreads) ∧
    (thread_state S (current_thread S) = Some tsRunning) ∧
    (spaceSpec ∈ Threads_Gno SysConf)
 then (if dIsPrivilegedSpace (GetThreadSpace S (current_thread S)) ∧
          spaceSpec ∈ threads S ∧
          (GetThreadSpace S spaceSpec) ∈ spaces S
       then (if cond
             then SetError (InitialiseAddressSpace S (GetThreadSpace S spaceSpec)) (current_thread S) eNoError
             else SetError S (current_thread S) (SOME e. e ∈ {eInvalidUtcbArea, eInvalidKipArea}))
       else SetError S (current_thread S) (SOME e. e ∈ {eNoPrivilege, eInvalidSpace}))
 else S)"

definition ExchangeRegister::" Sys_Config ⇒ State ⇒ globalid_t ⇒ control_t set⇒ nat ⇒ nat ⇒ nat ⇒ globalid_t ⇒ nat ⇒ State" where
"ExchangeRegister SysConf S gno control sp ip flags pager handle = 
(if (current_thread S ∈ active_threads S) ∧ (current_thread S ∉ kIntThreads) ∧ (thread_state S (current_thread S) = Some tsRunning) ∧
    (gno ∈ Threads_Gno SysConf) ∧ (pager ∈ Threads_Gno SysConf)
 then (if gno ∈ active_threads S ∧ (the (thread_space S gno) = GetCurrentSpace S)
       then(if cond
            then Weak_Exregs SysConf S gno control pager
            else SetError S (current_thread S) (SOME e. e ∈ {eOutOfMemory, eInvalidUtcbLocation}))
       else SetError S (current_thread S) eInvalidThread)
 else S)"

definition ThreadSwitch::"Sys_Config ⇒ State ⇒ threadid_t ⇒ State" where
"ThreadSwitch SysConf S destNo = 
(if (current_thread S ∈ active_threads S) ∧ (current_thread S ∉ kIntThreads) ∧ (thread_state S (current_thread S) = Some tsRunning) ∧
    (destNo ∈ GLOBAL_TID SysConf) ∧ destNo ≠ ANYTHREAD
 then (if destNo ≠ NILTHREAD
       then (if TidToGno destNo ∈ active_threads S ∧ thread_state S (TidToGno destNo) = Some tsRunning
                     ∧ TidToGno destNo ≠ current_thread S
                 then switch_to (enqueue_ready S (current_thread S)) (TidToGno destNo)
                 else S)
           else (let
                     S1 = enqueue_ready S (current_thread S);
                     S2 = S1⦇current_timeslice:=0⦈;
                     S3 = end_of_timeslice S2 (current_thread S);
                     S4 = Scheduling S3 (current_thread S)
                  in S4))
 else S)"

definition Schedule::"Sys_Config ⇒ State ⇒ threadid_t ⇒ int ⇒ nat ⇒ nat ⇒ nat ⇒ State" where
"Schedule SysConf S destNo timeControl procControl prio preemptControl = 
(if (current_thread S ∈ active_threads S) ∧ (current_thread S ∉ kIntThreads) ∧ (thread_state S (current_thread S) = Some tsRunning) ∧
    (destNo ∈ GLOBAL_TID SysConf)
 then (if TidToGno destNo ∉ active_threads S
          then SetError S (current_thread S) eInvalidThread
          else
      (if GetThreadScheduler S (TidToGno destNo) ∉ active_threads S
          then SetError S (current_thread S) eNoPrivilege
          else
      (if GetThreadSpace S (GetThreadScheduler S (TidToGno destNo)) ≠ GetThreadSpace  S (current_thread S)
          then SetError S (current_thread S) eNoPrivilege
          else do_schedule S (TidToGno destNo) timeControl prio)))
 else S)"

definition Unmap::"State ⇒ bool ⇒ message_t ⇒ State" where
"Unmap S control msg = 
(if (current_thread S ∈ threads S) ∧ (current_thread S ∈ active_threads S) ∧ (current_thread S ∉ kIntThreads) ∧ (thread_state S (current_thread S) = Some tsRunning)
then (let cur = current_thread S;
          space = GetCurrentSpace S
       in (case msg of UNTYPED n ⇒
            if fpage_type (nat_to_fpage n) = valid_fpage
            then
              let fpage = nat_to_fpage n;
                  f_base = fp_base fpage;
                  f_size = fp_size fpage;
                  f_perms = fp_perms fpage
               in (if control
                   then SetError (Flush_fpage S space f_base f_perms f_size) cur eNoError
                   else SetError (Unmap_fpage S space f_base f_perms f_size) cur eNoError)
            else SetError S cur eInvalidParameter
         |
          _ ⇒ SetError S cur eInvalidParameter))
else S)"
