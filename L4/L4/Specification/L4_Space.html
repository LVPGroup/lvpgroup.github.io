section‹ Address Space and Mapping›
definition "pgsize = (10::nat)"

definition "page_size = (2 ^ pgsize::vaddr_t)"

definition "pgmaxnum = 32 - pgsize"

definition "page_maxnum = (2 ^ pgmaxnum::vaddr_t)"

definition vaddr_to_vpage::"vaddr_t ⇒ v_page_t" where
"vaddr_to_vpage vaddr = (vaddr div page_size)"

definition vpage_to_vaddr::"v_page_t ⇒ vaddr_t" where
"vpage_to_vaddr vpage = vpage * page_size"

definition paddr_to_rpage::"paddr_t ⇒ r_page_t" where
"paddr_to_rpage paddr = (paddr div page_size)"

definition rpage_to_paddr::"r_page_t ⇒ paddr_t" where
"rpage_to_paddr rpage = rpage * page_size"

subsection‹ Fpage ›
definition fp_base::"fpage_t ⇒ nat" where
"fp_base fp = fst fp"

definition fp_size::"fpage_t ⇒ nat" where
"fp_size fp = fst (snd fp)"

definition fp_perms::"fpage_t ⇒ perms_t set" where
"fp_perms fp = snd (snd fp)"

definition fpage_type::"fpage_t ⇒ fpage_type" where
"fpage_type fp =
(if fp = (0,1,UNIV) then complete_fpage else
 if fp = (0,0,{})    then nil_fpage      else
 if fp_size fp ≥ 10 ∧ (fp_base fp * page_size) mod 2 ^ (fp_size fp) = 0 ∧ fp_perms fp ≠ {}
 then valid_fpage
 else invalid_fpage)"

definition nat_to_fpage::"nat ⇒ fpage_t" where
"nat_to_fpage n = (let base = n div page_size;
                       size = (n mod page_size) div 2^4;
                       perms = (if (n mod 2^3) div 2^2 > 0 then {pfRead} else {}) ∪
                               (if (n mod 2^2) div 2 > 0 then {pfWrite} else {}) ∪
                               (if (n mod 2^1) > 0 then {pfExecute} else {})
                     in (base,size,perms))"

"is_kSigma0Space s = (s = Sigma0Space)"

"is_kRootServerSpace s = (s = RootServerSpace)"

definition dIsPrivilegedSpace::"spaceName_t ⇒ bool" where
"dIsPrivilegedSpace s = (is_kSigma0Space s ∨ is_kRootServerSpace s)"

subsubsection‹ Path ›

definition get_perms::"State ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set" where
"get_perms s sp v_page = snd (the (the (space_mapping s sp) v_page))"

definition "get_perms1 s x ≡ case x of Virtual sp v ⇒ snd(the (the (space_mapping s sp) v))"

lemma get_perms_auxi:"x = Virtual sp v ⟹ get_perms s sp v = get_perms1 s x"
  unfolding get_perms_def get_perms1_def
  by simp

definition direct_path::"State ⇒ page_t ⇒ page_t ⇒ bool"  ("(_⊢_↝⇧1_)"[1000,1000,1000]) where
"s ⊢ p1 ↝⇧1 p2 = (∃sp v m perms. p1 = Virtual sp v ∧ space_mapping s sp = Some m ∧ m v = Some (p2,perms))"

lemma direct_def:"s ⊢ (Virtual sp v) ↝⇧1 p2 ⟶ (∃m perms. space_mapping s sp = Some m ∧ m v = Some (p2, perms))"
  by (simp add:direct_path_def)

lemma SonIsNoReal:"(s ⊢ p1 ↝⇧1 p2) ⟶ (∄r. p1 = Real r)"
  by (auto simp add:direct_path_def)

lemma FatherIsVirtual:"(s ⊢ p1 ↝⇧1 p2) ⟶ (∃sp v. p1 = Virtual sp v)"
  by (auto simp add:direct_path_def)

inductive tran_path::"State ⇒ page_t ⇒ page_t ⇒ bool" ("(_⊢_↝⇧+_)"[1000,1000,1000]) for s where
one_path:"s ⊢ x ↝⇧1 y ⟹ s ⊢ x ↝⇧+ y"|
tran_path:"s ⊢ x ↝⇧1 y ⟹ s ⊢ y ↝⇧+ z ⟹ s ⊢ x ↝⇧+ z"

inductive rtran_path::"State ⇒ page_t ⇒ page_t ⇒ bool" ("(_⊢_↝⇧*_)"[1000,1000,1000]) for s where
refl_path:"s ⊢ x ↝⇧* x"|
refl_tran:"s ⊢ x ↝⇧1 y ⟹ s ⊢ y ↝⇧* z ⟹ s ⊢ x ↝⇧* z"

lemma rtran_tran:
  "s ⊢ p1 ↝⇧* p2 ⟷ p1 = p2 ∨ s ⊢ p1 ↝⇧+ p2"
proof(rule iffI)
  show "s⊢p1↝⇧*p2 ⟹ p1 = p2 ∨ s⊢p1↝⇧+p2"
    apply(induct rule: rtran_path.induct)
    by(auto simp add:tran_path.intros)
next
  show "p1 = p2 ∨ s⊢p1↝⇧+p2 ⟹ s⊢p1↝⇧*p2"
    apply auto
     apply(rule rtran_path.intros)
    apply(induct rule:tran_path.induct)
    by(auto simp add:rtran_path.intros)
qed

subsubsection‹ Unmap ›
definition sys_unmap::"State ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set ⇒ State" where
"sys_unmap s sp v_page del_perms =
s⦇space_mapping:= 
    λsp'. (if space_mapping s sp' = None then None else
           Some (λv_page1. if the (space_mapping s sp') v_page1 = None then None else 
                                if s ⊢ (Virtual sp' v_page1) ↝⇧+ (Virtual sp v_page) 
                                then
                                        if snd (the (the (space_mapping s sp') v_page1)) ⊆ del_perms 
                                        then None
                                        else Some (fst (the (the (space_mapping s sp') v_page1)),
                                                   snd (the (the (space_mapping s sp') v_page1)) - del_perms)
                                else the (space_mapping s sp') v_page1
                      ))⦈"

subsubsection‹ Flush ›
definition sys_flush::"State ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set ⇒ State" where
"sys_flush s sp v_page del_perms ≡
if (sp ≠ Sigma0Space)
then s⦇space_mapping:= 
      λsp'. (if space_mapping s sp' = None then None else
             Some (λv_page1. if the (space_mapping s sp') v_page1 = None then None else 
                                  if s ⊢ (Virtual sp' v_page1) ↝⇧* (Virtual sp v_page) 
                                  then
                                          if snd (the (the (space_mapping s sp') v_page1)) ⊆ del_perms 
                                          then None
                                          else Some (fst (the (the (space_mapping s sp') v_page1)),
                                                     snd (the (the (space_mapping s sp') v_page1)) - del_perms)
                                  else the (space_mapping s sp') v_page1
                        ))⦈
else s"

subsubsection‹ Mapping ›
definition valid_page::"State ⇒ page_t ⇒ bool"  ("(_⊢_)" [1000,1000])where
"valid_page s page = (case page of Virtual sp v_page ⇒ (∃p. s ⊢ page ↝⇧1 p)| Real r ⇒ True)"

lemma FatherIsValid:"s ⊢ x ↝⇧1 y ⟶ s ⊢ x ∧ (∀r. x ≠ Real r)"
  unfolding direct_path_def valid_page_def
  by force

lemma ValidExSon:"s ⊢ x ∧ (∀r. x ≠ Real r) ⟶ (∃y. s ⊢ x ↝⇧1 y)"
  using direct_path_def valid_page_def page_t.simps(5) page_t.exhaust
  by smt

lemma FatherEqValid:"(∃y. s ⊢ x ↝⇧1 y) ⟷ s ⊢ x ∧ (∀r. x ≠ Real r)"
  using FatherIsValid ValidExSon by auto

lemma ValidInTran:"s ⊢ x ∧ (∀r. x ≠ Real r) ⟶ (∃y. s ⊢ x ↝⇧+ y)"
  using ValidExSon tran_path.intros(1)
  by blast

lemma ValidInRTran:"s ⊢ x ∧ (∀r. x ≠ Real r) ⟶ (∃y. s ⊢ x ↝⇧* y)"
  using ValidInTran rtran_tran
  by blast

lemma ValidVpageHasSon:"s ⊢ (Virtual sp vpage) = (∃y. s ⊢ (Virtual sp vpage) ↝⇧1 y)"
  using FatherEqValid
  by simp

lemma ValidVpageInTran:"x = Virtual sp vpage ⟹ s ⊢ x ⟶ (∃y. s ⊢ x ↝⇧+ y)"
  using ValidInTran
  by simp

lemma ValidVpageInRTran:"x = Virtual sp vpage ⟹ s ⊢ x ⟶ (∃y. s ⊢ x ↝⇧* y)"
  using ValidInRTran
  by simp

lemma ValidVpageMappingNotNone:"s ⊢ (Virtual sp vpage) = (space_mapping s sp ≠ None ∧ (the (space_mapping s sp)) vpage ≠ None)"
  using ValidVpageHasSon direct_path_def
  by auto

lemma ValidVpageMappingEx:"s ⊢ (Virtual sp vpage) = (∃m p2 perms. space_mapping s sp = Some m ∧ m vpage = Some (p2, perms))"
  using ValidVpageHasSon direct_path_def
  by auto

definition sys_map::"State ⇒ spaceName_t ⇒ v_page_t ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set ⇒ State" where
"sys_map s sp_from v_from sp_to v_to perms =
(if (sp_to ≠ Sigma0Space) 
 then 
    (if perms ≠ {} ∧ s ⊢ (Virtual sp_from v_from) ∧ perms ⊆ get_perms s sp_from v_from ∧ sp_from ≠ sp_to ∧
       (∀v. ¬s ⊢ (Virtual sp_from v_from) ↝⇧+ (Virtual sp_to v)) ∧ space_mapping s sp_to ≠ None ∧ v_to < page_maxnum
     then s⦇space_mapping:= 
      λsp'. (if space_mapping s sp' = None then None else
             Some (λv_page1. if the (space_mapping s sp') v_page1 = None ∧ 
                                Virtual sp' v_page1 ≠ Virtual sp_to v_to then None else 
                                    if s ⊢ (Virtual sp' v_page1) ↝⇧* (Virtual sp_to v_to) 
                                    then
                                          if (Virtual sp' v_page1) = (Virtual sp_to v_to)
                                          then Some ((Virtual sp_from v_from),perms)
                                          else None
                                    else the (space_mapping s sp') v_page1
                        ))⦈
     else s)
 else s)"

subsubsection‹ Grant ›
(*sp_to can not be sigma0*)
definition sys_grant::"State ⇒ spaceName_t ⇒ v_page_t ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set ⇒ State" where
"sys_grant s sp_from v_from sp_to v_to perms =
(if (sp_to ≠ Sigma0Space) 
 then 
    (if perms ≠ {} ∧ s ⊢ (Virtual sp_from v_from) ∧ sp_from ≠ sp_to ∧ perms ⊆ get_perms s sp_from v_from ∧ 
        (∀v. ¬s ⊢(fst (the (the (space_mapping s sp_from) v_from))) ↝⇧* (Virtual sp_to v)) ∧ 
         space_mapping s sp_to ≠ None ∧ v_to < page_maxnum
     then s⦇space_mapping:= 
      λsp'. (if space_mapping s sp' = None then None else
             Some (λv_page1. if the (space_mapping s sp') v_page1 = None ∧ 
                                Virtual sp' v_page1 ≠ Virtual sp_to v_to then None else 
                                  if s ⊢ (Virtual sp' v_page1) ↝⇧* (Virtual sp_to v_to) ∨
                                     s ⊢ (Virtual sp' v_page1) ↝⇧* (Virtual sp_from v_from) 
                                  then
                                        if (Virtual sp' v_page1) = (Virtual sp_to v_to)
                                        then Some (fst (the (the (space_mapping s sp_from) v_from)),perms)
                                        else None
                                  else the (space_mapping s sp') v_page1
                        ))⦈
     else s)
 else s)"

subsection‹ Important Function›
fun Map_fpage::"State ⇒ spaceName_t ⇒ v_page_t ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set ⇒ nat ⇒ State" where
"Map_fpage s sp1 v1 sp2 v2 perms 0 = s"|
"Map_fpage s sp1 v1 sp2 v2 perms (Suc n) =
    (sys_map (Map_fpage s sp1 v1 sp2 v2 perms n) sp1 (v1+n) sp2 (v2+n) perms)"

fun Grant_fpage::"State ⇒ spaceName_t ⇒ v_page_t ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set ⇒ nat ⇒ State" where
"Grant_fpage s sp1 v1 sp2 v2 perms 0 = s"|
"Grant_fpage s sp1 v1 sp2 v2 perms (Suc n) =
    (sys_grant (Grant_fpage s sp1 v1 sp2 v2 perms n) sp1 (v1+n) sp2 (v2+n) perms)"

fun Unmap_fpage::"State ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set ⇒ nat ⇒ State" where
"Unmap_fpage s sp v perms 0 = s"|
"Unmap_fpage s sp v perms (Suc n) =
    (sys_unmap (Unmap_fpage s sp v perms n) sp (v+n) perms)"

fun Flush_fpage::"State ⇒ spaceName_t ⇒ v_page_t ⇒ perms_t set ⇒ nat ⇒ State" where
"Flush_fpage s sp v perms 0 = s"|
"Flush_fpage s sp v perms (Suc n) =
    (sys_flush (Flush_fpage s sp v perms n) sp (v+n) perms)"

subsection‹ Create Space ›
definition CreateAddressSpace::"Sys_Config ⇒ State ⇒ spaceName_t ⇒ State" where
"CreateAddressSpace config s sp = 
(if sp ∈ (Address_Space config) ∧ sp ∉ spaces s 
 then s⦇
        space_threads:= (space_threads s)(sp := Some {}),
        space_mapping:= (space_mapping s)(sp := Some (λv_page. None))⦈
 else s)"

subsection‹ Initialise Space ›
definition InitialiseAddressSpace::"State ⇒ spaceName_t ⇒ State" where
"InitialiseAddressSpace s sp =
(if sp ∈ spaces s 
then s⦇initialised_spaces:= (initialised_spaces s ∪ {sp})⦈
else s)"

subsection‹ Delete Space ›
definition DeleteAddressSpace::"State ⇒ spaceName_t ⇒ State" where
"DeleteAddressSpace s sp =
(if sp ∈ spaces s ∧ ¬ dIsPrivilegedSpace sp 
then 
  let s1 = Unmap_fpage s sp 0 UNIV page_maxnum   ― ‹Unmap everything; complete fpage›
   in s1⦇
         initialised_spaces:= initialised_spaces s1 - {sp},
         space_threads:= (space_threads s1)(sp:= None),
         space_mapping:= (space_mapping s1)(sp:= None)⦈
else s)"
