theory L4_Inv_Proof
  imports L4_Inv_WeakSyscall

section‹Invariant Proofs›

lemma s0_Inv:"Invariants s0_req"
  unfolding Invariants_def
  using s0_Inv_Current s0_Inv_Space s0_Inv_Thread s0_Inv_Schedule s0_Inv_IPC
  by auto

lemma IPC_Inv:
  assumes p1:"Invariants s"
      and p2:"s1 = IPC SysConf s to_gno fromSpecifier send_timeout recv_timeout"
    shows "Invariants s1"
  unfolding p2 IPC_def
  using p1 SetError_Inv WeakIPC_Inv
  by auto

  assumes p1:"Invariants s"
      and p2:"s1 = ThreadControl SysConf s destNo spaceSpec schedNo pagerNo"
    shows "Invariants s1"
  unfolding ThreadControl_def p2
  using p1 ThreadControl_Cond_def WeakCreateThread_Inv
    WeakModifyThread_Inv WeakDeleteThread_Inv IntThreadControl_Inv SetError_Inv
  by simp

lemma ThreadSwitch_Inv:
  assumes p1:"Invariants s"
      and p2:"s1 = ThreadSwitch SysConf s dest"
    shows "Invariants s1"
  unfolding p2 ThreadSwitch_def
  apply(rule elim_if)
  subgoal
    apply(rule elim_if)
    subgoal
      apply(rule elim_if)
      subgoal
        apply(subst switch_to_Inv)
        subgoal
          using enqueue_ready_NC_Thread by auto
        subgoal
          using enqueue_ready_Inv p1 by auto
        by simp
      using p1 by simp
    unfolding Let_def
    apply(subst Scheduling_Inv)
    subgoal
      apply(subst end_of_timeslice_NC_Thread)
      using enqueue_ready_NC_Thread by auto
    subgoal
      apply(subst end_of_timeslice_Inv)
      subgoal
        using enqueue_ready_NC_Thread by auto
      subgoal
        apply(subst SetCurrentTimeslice_Inv[unfolded SetCurrentTimeslice_def,where t = "0"])
        subgoal
          using enqueue_ready_Inv p1 by auto
        by simp
      by simp
    by simp
  using p1 by simp

lemma Schedule_Inv:
  assumes p1:"Invariants s"
      and p2:"s1 = Schedule SysConf s destNo timeControl procControl prio preemptControl"
    shows "Invariants s1"
  unfolding p2 Schedule_def 
  using SetError_Inv do_schedule_Inv p1
  by simp

lemma Unmap_Inv:
  assumes p1:"Invariants s"
  shows "Invariants (Unmap s control msg)"
  unfolding Unmap_def Let_def
  apply(rule elim_if_and_pre)
  using p1 SetError_Inv
   apply(cases msg)
      apply auto
    apply(subst SetError_Inv)
     apply auto
  using Flush_fpage_Invariant
    apply simp
   apply(subst SetError_Inv)
    apply auto
  using Unmap_fpage_Invariant p1
   apply auto
  done

lemma step_Inv:
  assumes p1:"Invariants s"
      and p2:"s' = step s a"
    shows "Invariants s'"
proof(cases a)
  case (syscall x1)
  then have h11:"a = syscall x1" by simp
  then show ?thesis
  proof(cases x1)
    case (eIPC x11 x12 x13 x14)
    then show ?thesis using p1 p2 step_def h11 IPC_Inv
      by simp
  next
    case (eThreadControl x21 x22 x23 x24)
    then show ?thesis using p1 p2 step_def h11 ThreadControl_Inv
      by simp
  next
    case (eSpaceControl x31 x32 x33 x34)
    then show ?thesis using p1 p2 step_def h11 SpaceControl_Inv
      by simp
  next
    case (eExchangeRegister x41 x42 x43 x44 x45 x46 x47)
    then show ?thesis using p1 p2 step_def h11 ExchangeRegister_Inv
      by simp
  next
    case (eThreadSwitch x5)
    then show ?thesis using p1 p2 step_def h11 ThreadSwitch_Inv
      by simp
  next
    case (eSchedule x61 x62 x63 x64 x65)
    then show ?thesis using p1 p2 step_def h11 Schedule_Inv
      by simp
  next
    case (eUnmap x7)
    then show ?thesis using p1 p2 step_def h11 Unmap_Inv
      by simp
  next
    case (eProcessorControl x81 x82 x83 x84)
    then show ?thesis using p1 p2 step_def h11 ProcessorControl_Inv
      by simp
  next
    case (eMemoryControl x91 x92 x93 x94)
    then show ?thesis using p1 p2 step_def h11 MemoryControl_Inv
      by simp
  next
    case eSystemClock
    then show ?thesis using p1 p2 step_def h11 SystemClock_Inv
      by simp
  qed
next
  case timer
  then show ?thesis using p1 p2 Timer_Inv step_def
    by simp
next
  case (interrupt x3)
  then show ?thesis using p1 p2 step_def
    by simp
qed

lemma Run_Inv:
  assumes "Invariants s"
  shows "Invariants (run s es)"
  using assms
proof(induction es arbitrary:s)
  case Nil
  then show ?case by simp
next
  case (Cons a es)
  then have "Invariants (step s a)" using step_Inv by simp
  then have "Invariants (run (step s a) es)" using Cons by simp
  then show ?case by simp
qed

lemma Reach_Inv:
  assumes a1:"Reachable s"
  shows "Invariants s"
proof-
  have h1:"∃es. s = run s0_req es" using assms Reachable_def by auto
  then obtain es where h2:"s = run s0_req es" by blast
  then show ?thesis using s0_Inv Run_Inv by auto
qed
