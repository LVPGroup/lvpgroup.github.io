theory L4_Basic_Lemmas
  imports "../specification/L4_Inv_Def"
begin
lemma elim_assm':"Q ⟹ (P ⟹ Q)"
  by simp

lemma elim_assm:"Q ⟹ P ⟶ Q"
  by simp

lemma elim_assms1:"A ⟶ C ⟹ A ∧ B ⟶ C"
  by simp

lemma elim_assms2:"B ⟶ C ⟹ A ∧ B ⟶ C"
  by simp
  thm if_split
lemma elim_if_and_pre:"⟦P x;P y⟧ ⟹ P (if (Q) then (x) else (y))"
  by simp

lemma elim_if:"(Q ⟹ P x) ⟹ (¬Q ⟹ P y) ⟹ P (if Q then x else y)"
  by simp

lemma elim_if_in_pre:"(a ∧ P b ⟹ Q) ⟹ (¬a ∧ P c ⟹ Q) ⟹ P (if a then b else c) ⟹ Q"
  by presburger

thm option.case
thm option.case_eq_if[no_vars]
thm prod.case_eq_if[no_vars]
thm if_split[no_vars]
thm option.the_def
not_None_eq
not_Some_eq

lemma elim_case_option:
"(option = None ⟹ P f1) ⟹ ((option ≠ None) ⟹ P (f2 (the option))) ⟹ P (case option of None ⇒ f1 | Some x ⇒ f2 x)"
  by (auto simp:option.case_eq_if)

lemma elim_case_option_ex:
"(option = None ⟹ P f1) ⟹ (⋀x. (option = Some x) ⟹ P (f2 (the option))) ⟹ P (case option of None ⇒ f1 | Some x ⇒ f2 x)"
  by (auto simp:option.case_eq_if)

lemma elim_case_prod:
"(option = None ⟹ P f1) ⟹ ((option ≠ None) ⟹ P (f2 (fst (the option)) (snd (the option)))) ⟹ P (case option of None ⇒ f1 | Some (a1,a2) ⇒ f2 a1 a2)"
  by (auto simp:option.case_eq_if)

lemma if_respect_eq:"(P s1 = P s) ⟹(P s2 = P s) ⟹ P (if p then s1 else s2) = P s"
  by simp

lemma if_respect_eq_imply:"(p ⟹ P s1 = P s) ⟹(¬p ⟹ P s2 = P s) ⟹ P (if p then s1 else s2) = P s"
  by simp

lemma if_respect_vpeq:"(P s1 s) ⟹(P s2 s) ⟹ P (if p then s1 else s2) s"
  by simp

lemma if_respect_vpeq_imply:"(p ⟹ P s1 s) ⟹(¬p ⟹ P s2 s) ⟹ P (if p then s1 else s2) s"
  by simp

lemma if_step_eq:"p1 = p2 ⟹ P s1 = P t1 ⟹ P s2 = P t2 ⟹ 
      P (if p1 then s1 else s2) = P (if p2 then t1 else t2)"
  by simp

lemma if_step_eq_imply:"p1 = p2 ⟹ (p1 ⟹ P s1 = P t1) ⟹ (¬p1 ⟹P s2 = P t2) ⟹ 
      P (if p1 then s1 else s2) = P (if p2 then t1 else t2)"
  by simp

lemma if_step_vpeq:"p1 = p2 ⟹ P s1 t1 ⟹ P s2 t2 ⟹ 
      P (if p1 then s1 else s2) (if p2 then t1 else t2)"
  by simp

lemma if_step_vpeq_imply:"p1 = p2 ⟹ (p1 ⟹ P s1 t1) ⟹ (¬p1 ⟹ P s2 t2) ⟹ 
      P (if p1 then s1 else s2) (if p2 then t1 else t2)"
  by simp

lemma let_respect_eq:"P (f s) = P s ⟹(⋀t. P (g t) = P t) ⟹ P (let s1 = f s in g s1) = P s"
  by simp

lemma let_step_eq:"P s = P t ⟹ (⋀ s t. P s = P t ⟹ P (f s) = P (f t)) ⟹ P (Let s f) = P (Let t f)"
  unfolding Let_def
  by blast

lemma let_step_vpeq:"P s t ⟹ (⋀ s t. P s t ⟹ P (f s) (f t)) ⟹ P (Let s f) (Let t f)"
  by simp

lemma if_vpeq_memory:"p1 = p2 ⟹ vpeq_memory s1 d t1 ⟹ vpeq_memory s2 d t2 ⟹
vpeq_memory (if p1 then s1 else s2) d (if p2 then t1 else t2)"
  by simp

lemma if_vpeq_memory_imply:"p1 = p2 ⟹ p1 ⟶ vpeq_memory s1 d t1 ⟹ ¬p1 ⟶ vpeq_memory s2 d t2 ⟹
vpeq_memory (if p1 then s1 else s2) d (if p2 then t1 else t2)"
  by simp

lemma let_vpeq_memory:"vpeq_memory s d t ⟹ ∀s t. vpeq_memory s d t ⟶ vpeq_memory (f s) d (f t) ⟹
vpeq_memory (Let s f) d (Let t f)"
  unfolding Let_def
  by auto
end
