theory PP_RSM
  imports Main 
          "../SPD/PP_DM"
          "../SPD/PP_TSM"
          "../SOs/PP_OSM"
begin


section "SFR State Machine"

type_synonym tee_id = nat

record 'cid OB_TA_STORAGE =
      sowner :: "'cid"
      inExtMem::bool
      TEE_identity::tee_id


record OB_SRT =
      TEE_identity2::tee_id

(*RSM*)

locale State = OState s0 Trans  getSoc getTS TA_storage TEE_persist  TEE_state TEE_state_abort
  for
      s0 :: 's (* initial state of the system *)
  and Trans :: "'s ⇒ 'a ⇒ 's" (* state machine of the whole system *)
  and TA_storage ::"'s⇒'ta⇒'ts"
  and TEE_persist::"'s⇒'tee_data"
  and TEE_state :: "'s⇒'tee_state"
  and TEE_state_abort :: "'tee_state"
  and getSoc::"'s⇒OB_SRT"
  and getTS::"'s⇒'ts"
  +
 
  fixes TAs :: "'s ⇒ 'ta set" (* deployed TAs on the TEE *)
  fixes CAs :: "'s ⇒ 'ca set" (* all CAs now are communicating with some TA *)
  fixes TA_state :: "'s ⇒ 'ta ⇒ 'ta_state"
 
  fixes TA_checkcode :: "'s⇒'k"(* the key of a TA code to check authenticity *)
  fixes TA_bincode :: "'s ⇒ 'ta ⇒ 'bc" (* TA bincode at runtime in the memory *)
  (*fixes Abort::"'s⇒'ta⇒'ta_state"*)
  fixes TA_state_abort :: 'ta_state (* a special state of abort *)
  fixes TA_instance_time :: "'s ⇒ 'ta ⇒ 'instime::linorder" (* TA's instance time *)
  fixes Audit_Records :: "'s⇒'ar"
 
begin

(*Lemma 1 (State machine refinement)*)
interpretation OState 
        done
end



type_synonym cid = nat
type_synonym tid = nat
type_synonym offset = nat
type_synonym binary = string
type_synonym key = string
type_synonym domain_id = nat
type_synonym ta_name = string
type_synonym session_id = nat
type_synonym uuid = nat

datatype TA_mode_type = Dead | Cold_Start | Normal
datatype TA_session_mode = Closed | Running

record TA_Session_Type = 
                server :: uuid
                session_mode :: TA_session_mode

datatype TA_conf = TAconf tid offset binary "TA_Session_Type list"

record Sys_Config = 
                    REE :: domain_id 
                    TEE :: domain_id
                    TAConf :: "tid⇀TA_conf" 
                    checkcode :: string

consts sysconf :: "Sys_Config" 


record TA_State_Type = 
                TA_mode :: TA_mode_type
                TA_sessions :: "session_id⇀TA_Session_Type"


type_synonym TAs_State = "tid⇀TA_State_Type"


record Statex = 
        current :: domain_id
        TAs :: TAs_State
        system_time :: nat
        
       
definition getTAbyId::"Statex⇒tid⇒TA_State_Type"where
        "getTAbyId s tid ≡the (TAs s tid)"

definition getIdbyTA::"Statex⇒TA_State_Type⇒tid"where
        "getIdbyTA s ta ≡ SOME tid1. getTAbyId s tid1 = ta"

definition getTAStateByid::"Statex⇒tid⇒TA_mode_type"where
  "getTAStateByid s tid ≡ TA_mode (the (TAs s tid))"
       
definition Abort::"Statex⇒tid⇒Statex"where
        "Abort s tid ≡ let ta = TAs s; tastate = the (ta tid) in s⦇TAs := ta(tid := Some (tastate⦇TA_mode := Dead⦈))⦈"

definition Close_Instance::"Statex⇒tid⇒Statex"where
        "Close_Instance s tid ≡ let ta = TAs s; tastate = the (ta tid) in s⦇TAs := ta(tid := Some (tastate⦇TA_mode := Dead⦈))⦈"

definition Close_Session::"Statex⇒tid⇒session_id⇒Statex"where
        "Close_Session s tid sid ≡ let ta = TAs s; tastate = the (ta tid); tasession = TA_sessions tastate; sesstate = the (tasession sid) 
           in s⦇TAs := ta(tid := Some (tastate⦇TA_sessions := tasession(sid := Some  (sesstate⦇session_mode := Closed⦈))⦈))⦈"

primrec getTAOffset :: "TA_conf⇒offset" where
      "getTAOffset (TAconf _ set1  _ _) = set1"

primrec getTAbinary :: "TA_conf⇒binary" where
      "getTAbinary (TAconf _ _  bin1 _) = bin1"

definition getTAInstanceTime::"Sys_Config⇒Statex⇒tid⇒nat"where
  "getTAInstanceTime sc s tid ≡ system_time s + getTAOffset (the (TAConf sc tid))" 

definition getTABinaryCode::"Sys_Config⇒Statex⇒tid⇒binary"where
"getTABinaryCode sc s tid ≡ getTAbinary (the (TAConf sc tid))"

definition getCheckCode::"Sys_Config⇒key"where
  "getCheckCode s ≡ checkcode s"


end
