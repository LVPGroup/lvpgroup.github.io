section "TEEC Client APIs"

definition TEEC_RegisterSharedMemory:: "Sys_Config ⇒ State ⇒ TEEC_CONTEXT_TYPE ⇒ TEEC_SharedMemory ⇒ (State × TEEC_RETURN_CODE_TYPE)" where
  "TEEC_RegisterSharedMemory sc s ctx shm ≡ 
  if current s ≠ REE sc then (s, TEEC_ERROR_BUSY)
  else 
    if (exec_prime s) ≠ [] then (s, TEEC_ERROR_BUSY)
    else 
      if (flags shm) ≠ TEEC_MEM_INPUT ∨ (flags shm) ≠ TEEC_MEM_OUTPUT then (s, TEEC_ERROR_BAD_PARAMETERS)
      else 
        let newblk = shm_alloc s (shared_size shm);
            newblk1 = newblk⦇related := Some (buffer shm)⦈;
            lst = newblk1 # driver_mem(REE_state s);
            newREESize = ree_total_size(REE_state s) - (shared_size shm)  
        in  (s⦇REE_state := (REE_state s)⦇driver_mem := lst, ree_total_size := newREESize⦈⦈,  TEEC_SUCCESS)
"

definition TEEC_AllocateSharedMemory:: "Sys_Config ⇒ State ⇒ TEEC_CONTEXT_TYPE ⇒ TEEC_SharedMemory ⇒ (State × TEEC_RETURN_CODE_TYPE)" where
  "TEEC_AllocateSharedMemory sc s ctx shm ≡ 
  if current s ≠ REE sc then (s, TEEC_ERROR_BUSY)
    else 
    if (exec_prime s) ≠ [] then (s, TEEC_ERROR_BUSY)
    else
      if (flags shm) ≠ TEEC_MEM_INPUT ∨ (flags shm) ≠ TEEC_MEM_OUTPUT then (s, TEEC_ERROR_BAD_PARAMETERS)
      else 
        let newblk = shm_alloc s (shared_size shm);
            newblk1 = newblk⦇related := Some 0⦈;
            lst = newblk1 # driver_mem(REE_state s);
            newREESize = ree_total_size(REE_state s) - (shared_size shm)  
        in  (s⦇REE_state := (REE_state s)⦇driver_mem := lst, ree_total_size := newREESize⦈⦈,  TEEC_SUCCESS)
"

definition TEEC_ReleaseSharedMemory :: "Sys_Config ⇒ State ⇒TEEC_SharedMemory ⇒ State " where
  "TEEC_ReleaseSharedMemory sc s shm ≡ 
  if current s ≠ REE sc then s
  else if (exec_prime s) ≠ [] then s
  else if shared_id shm = 0 then s
  else  let newREESize = ree_total_size(REE_state s) + (alloced_size shm);
            newlst = filter (λ(x::MemBlock). (block_id x) ≠ (buffer shm)) (driver_mem(REE_state s)) 
        in  s⦇REE_state := (REE_state s)⦇ree_total_size := newREESize, driver_mem := newlst⦈⦈"

definition TEE_CheckMemoryAccessRights :: "Sys_Config ⇒ State ⇒ accessFlags ⇒ MemBlock ⇒ BUFFER_SIZE_TYPE ⇒ TEE_RETURN_CODE_TYPE" where
  "TEE_CheckMemoryAccessRights sc s aF memblock  buffer_size ≡ 
    if current s = TEE sc ∨ current s = REE sc then TEE_ERROR_BUSY
    else if (exec_prime s) ≠ [] then TEE_ERROR_BUSY
    else if buffer_size = 0 then TEE_SUCCESS
    else if (aF = TEE_MEMORY_ACCESS_SECURE ∧ (is_SecureMem memblock) = True)
           ∨(aF = TEE_MEMORY_ACCESS_NONSECURE ∧ (is_SecureMem memblock) = False)
           ∨(aF = TEE_MEMORY_ACCESS_WRITE ∧ (access_right memblock (current s)) = Some WRITE)
           ∨(aF = TEE_MEMORY_ACCESS_READ ∧ (access_right memblock (current s)) = Some READ)
           ∨(aF = TEE_MEMORY_ACCESS_ANY_OWNER  ∧ access_right memblock = Map.empty)
         then TEE_SUCCESS
         else TEE_ERROR_GENERIC"

definition TEE_Malloc :: "Sys_Config ⇒ State ⇒ BUFFER_SIZE_TYPE ⇒ hint ⇒ State" where
  "TEE_Malloc sc s buffer_size h ≡ 
    if current s = TEE sc ∨ current s = REE sc then s
    else if (exec_prime s) ≠ [] then s
    else if ((h ≠ TEE_MALLOC_FILL_ZERO) ∧ (h ≠ TEE_USER_MEM_HINT_NO_FILL_ZERO)) then s
    else 
         let newTEESize = tee_total_size(TEE_state s) - buffer_size;
             newblk = mem_alloc s buffer_size;
             lst = newblk # tee_memories(TEE_state s);
             newBIDpointer = BIDpointer(ta_mgr(TEE_state s)) + 1
         in s⦇TEE_state := (TEE_state s)⦇tee_total_size := newTEESize, tee_memories := lst, ta_mgr:=ta_mgr(TEE_state s)⦇BIDpointer:=newBIDpointer⦈⦈⦈"

definition TEE_Realloc :: "Sys_Config ⇒ State ⇒ MemBlock ⇒ BUFFER_SIZE_TYPE ⇒ State" where
  "TEE_Realloc sc s memblock new_size ≡
    if current s = TEE sc ∨ current s = REE sc then s
    else if (exec_prime s) ≠ [] then s
    else let newTEESize = tee_total_size(TEE_state s) - new_size + (size memblock)
         in  s⦇TEE_state := (TEE_state s)⦇tee_total_size := newTEESize⦈⦈"

definition TEE_Free :: "Sys_Config ⇒ State ⇒ MemBlock ⇒ State" where
  "TEE_Free sc s memblock ≡
    if current s = TEE sc ∨ current s = REE sc then s
    else if (exec_prime s) ≠ []
         ∨ ownership memblock ≠ current s
         ∨ block_id memblock < 2
         ∨ memblock ∉ set(filter (λx.(ownership x) = (ownership memblock)) (tee_memories(TEE_state s)))
         then s
    else
          let newTEESize = tee_total_size(TEE_state s) + (size memblock);
              newlst = filter (λx.(block_id x) ≠ (block_id memblock)) (tee_memories(TEE_state s))
          in  s⦇TEE_state := (TEE_state s)⦇tee_total_size := newTEESize, tee_memories := newlst⦈⦈"
