  "tee_ta_invoke_command sc s ses_id time_out cmd_id in_params out_params ≡
    let
        client_tid = the(findSesCliTid s ses_id);
        server_tid = the(findSesServTid s ses_id);
        taInsList = mgr_ta_instances (ta_mgr (TEE_state s));
        client_taIns = getTAInsCtx s client_tid;
        server_taIns = getTAInsCtx s server_tid;
        clnt_id = ta_id client_taIns;
        taMgrSes = the(getTASessionBySesIdFromMgr s ses_id);
        curSesClintId = the(id (client_id taMgrSes))
    in
        if clnt_id ≠ curSesClintId then
          (s, TEE_ORIGIN_TEE, TEE_ERROR_BAD_PARAMETERS, out_params)
        else
           let 
                isServTaBusy = busy server_taIns
           in
              if isServTaBusy = True then
                (s, TEE_ORIGIN_TEE, TEE_ERROR_BAD_PARAMETERS, out_params)
              else
                let
                    s_setServTaInsBusy = setTaInsBusyByThreadId s server_tid True;
                    ret_invokeServTaByThreadId = invokeServTaByThreadId sc s_setServTaInsBusy ses_id cmd_id in_params out_params;
                    s_invokeServTaByThreadId = fst ret_invokeServTaByThreadId;
                    retCode_invokeServTaByThreadId = snd ret_invokeServTaByThreadId
                in
                  if retCode_invokeServTaByThreadId ≠ TEE_SUCCESS then
                    (s_invokeServTaByThreadId, DefaultOriginRet, retCode_invokeServTaByThreadId, out_params)
                  else
                    let
                      s_unsetTaInsBusyByThreadId = setTaInsBusyByThreadId s_invokeServTaByThreadId server_tid False
                    in
                    (s_unsetTaInsBusyByThreadId, DefaultOriginRet, TEE_ERROR_BAD_PARAMETERS, out_params)"

definition tee_invokeTACommand ::
  "Sys_Config ⇒ State ⇒ SESSION_ID_TYPE option ⇒ TIMEOUT_TYPE ⇒ COMMAND_ID_TYPE ⇒ PARAMS_TYPE ⇒ PARAMS_TYPE
  ⇒ State × TEE_RETURN_ORIGIN_TYPE × TEE_RETURN_CODE_TYPE × PARAMS_TYPE" where
  "tee_invokeTACommand sc s ses_id time_out cmd_id in_params out_params ≡
  if ses_id = None then (s, DefaultOriginRet, TEE_ERROR_BAD_PARAMETERS, out_params)
  else
    let
        ses_id = the ses_id;
        clientTid = findSesCliTid s ses_id;
        servTid = findSesServTid s ses_id
    in
        if clientTid = None ∨ servTid = None then
          (s, DefaultOriginRet, TEE_ERROR_BAD_PARAMETERS, out_params)
        else 
          let
            taInsList = mgr_ta_instances (ta_mgr (TEE_state s));
            client_taIns = getTAInsCtx s (the clientTid);
            clnt_id = ta_id client_taIns;
            taMgrSes = the(getTASessionBySesIdFromMgr s ses_id);
            curSesClintId = the(id (client_id taMgrSes))
          in
            if clnt_id ≠ curSesClintId then
              (s, DefaultOriginRet, TEE_ERROR_BAD_PARAMETERS, out_params)
            else
                let
                  ret_invokeCmd = tee_ta_invoke_command sc s ses_id time_out cmd_id in_params out_params;
                  s_invokeCmd = fst ret_invokeCmd;
                  ret_ori = fst (snd ret_invokeCmd);
                  ret_code = fst (snd (snd ret_invokeCmd));
                  ret_params = (snd (snd (snd ret_invokeCmd)))
                in
                  if ret_code ≠ TEE_SUCCESS then
                    (s_invokeCmd, ret_ori, ret_code, out_params)
                  else
                    (s_invokeCmd, TEE_ORIGIN_TRUSTED_APP, TEE_SUCCESS, out_params)"
